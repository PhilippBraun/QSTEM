77 void Detector::CollectIntensity(const WavePtr &wave)
78 {
79  /********************************************************************
80  * collectIntensity(muls, wave, slice)
81  * collect the STEM signal on the annular detector(s) defined in muls
82  * and write the appropriate pixel in the image for each detector and thickness
83  * The number of images is determined by the following formula:
84  * muls->slices*muls->cellDiv/muls->outputInterval
85  * There are muls->detectorNum different detectors
86  *******************************************************************/
87  int ixs,iys,t;
88  float_tt k2;
89  float_tt intensity,scale,scaleCBED,scaleDiff,intensity_save;
90  char fileName[256],avgName[256];
91  float_tt **diffpatAvg = NULL;
92  int tCount = 0;
93  unsigned nx, ny, offsetX, offsetY;
94 
95  std::vector<std::vector<DetectorPtr> > detectors;
96 
97  wave->GetSizePixels(nx, ny);
98  wave->GetPositionOffset(offsetX, offsetY);
99 
100  //scale=1;
101  // TODO: come up with where electronScale belongs (might be STEM-specific?
102  scale = m_electronScale/((double)(nx*ny)*(nx*ny));
103  // scaleCBED = 1.0/(scale*sqrt((double)(muls->nx*muls->ny)));
104  scaleDiff = 1.0/sqrt((double)(nx*ny));
105 
106  int position_offset = offsetY * m_nx + offsetX;
107 
108  // Multiply each image by its number of averages and divide by it later again:
109  m_image[position_offset] *= m_Navg;
110  m_image2[position_offset] *= m_Navg;
111  m_error = 0;
112 
113  /* add the intensities in the already
114  fourier transformed wave function */
115  unsigned px = nx*ny;
116  for (unsigned i = 0; i < px; i++)
117  {
118  {
119  unsigned ix = i%nx;
120  unsigned iy = i/nx;
121  k2 = wave->GetK2(ix, iy);
122  intensity = wave->GetPixelIntensity(i);
123  wave->SetDiffPatPixel((ix+nx/2)%nx,(iy+ny/2)%ny, intensity*scaleDiff);
124  intensity *= scale;
125  if ((k2 >= m_k2Inside) && (k2 <= m_k2Outside))
126  {
127  // detector in center of diffraction pattern:
128  if ((m_shiftX == 0) && (m_shiftY == 0))
129  {
130  m_image[position_offset] += intensity;
131  // misuse the error number for collecting this pixels raw intensity
132  m_error += intensity;
133  }
134  /* special case for shifted detectors: */
135  else
136  {
137  intensity_save = intensity;
138  ixs = (ix+(int)m_shiftX+nx) % nx;
139  iys = (iy+(int)m_shiftY+ny) % ny;
140  intensity = scale * wave->GetPixelIntensity(ixs, iys);
141  m_image[position_offset] += intensity;
142  // repurpose the error number for collecting this pixels raw intensity
143  m_error += intensity;
144  /* restore intensity, so that it will not be shifted for the other detectors */
145  intensity = intensity_save;
146  }
147  } /* end of if k2 ... */
148  } /* end of for iy=0... */
149  } /* end of for ix = ... */
150 
151 
152 
153  // Divide each image by its number of averages again:
154  // add intensity squared to image2 for this detector and pixel, then rescale:
155  m_image2[position_offset] += m_error*m_error;
156  m_image2[position_offset] /= m_Navg+1;
157 
158  // do the rescaling for the average image:
159  m_image[position_offset] /= m_Navg+1;
160 }
