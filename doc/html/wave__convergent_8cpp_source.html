106 /**********************************************
107 * This function creates a incident STEM probe
108 * at position (dx,dy)
109 * with parameters given in muls
110 *
111 * The following Abberation functions are being used:
112 * 1) ddf = Cc*dE/E + Cc2*(dE/E)^2,
113 * Cc, Cc2 = chrom. Abber. (1st, 2nd order) [1]
114 * 2) chi(qx,qy) = (2*pi/lambda)*{0.5*C1*(qx^2+qy^2)+
115 * 0.5*C12a*(qx^2-qy^2)+
116 * C12b*qx*qy+
117 * C21a/3*qx*(qx^2+qy^2)+
118 * ...
119 * +0.5*C3*(qx^2+qy^2)^2
120 * +0.125*C5*(qx^2+qy^2)^3
121 * ... (need to finish)
122 *
123 *
124 * qx = acos(kx/K), qy = acos(ky/K)
125 *
126 * References:
127 * [1] J. Zach, M. Haider,
128 * "Correction of spherical and Chromatic Abberation
129 * in a low Voltage SEM", Optik 98 (3), 112-118 (1995)
130 * [2] O.L. Krivanek, N. Delby, A.R. Lupini,
131 * "Towards sub-Angstroem Electron Beams",
132 * Ultramicroscopy 78, 1-11 (1999)
133 *
134 *********************************************/
135 #define SMOOTH_EDGE 5 // make a smooth edge on AIS aperture over +/-SMOOTH_EDGE pixels
136 void CConvergentWave::FormProbe()
137 {
138  // static char *plotFile = "probePlot.dat",systStr[32];
139  unsigned ix, iy, ixmid, iymid;
141  float_tt rmin, rmax, aimin, aimax;
142  // float **pixr, **pixi;
143  float_tt k2max, x, y, scale, pixel,alpha, r;
144  //double envelope;
145 
146  // FILE *fp=NULL;
147 
148  /* temporary fix, necessary, because fftw has rec. space zero
149  in center of image:
150  */
151  float_tt ax = m_nx*m_dx;
152  float_tt by = m_ny*m_dy;
153  // these are offsets of some kind:
154 
155  float_tt dx = ax-m_nx/2*m_dx;
156  float_tt dy = by-m_ny/2*m_dy;
157 
158  // average resolution:
159  float_tt avgRes = sqrt(0.5*(m_dx*m_dx+m_dy*m_dy));
160  float_tt edge = SMOOTH_EDGE*avgRes;
161 
162  /********************************************************
163  * formulas from:
164  * http://cimesg1.epfl.ch/CIOL/asu94/ICT_8.html
165  *
166  * dE_E = dE/E = energy spread of emitted electrons
167  * dV_V = dV/V = acc. voltage fluctuations
168  * dI_I = dI/I = lens current fluctuations
169  * delta defocus in Angstroem (Cc in A)
170  *******************************************************/
171  float_tt delta = m_Cc*m_dE_E;
172  if (m_printLevel > 2) printf("defocus offset: %g nm (Cc = %g)\n",delta,m_Cc);
173 
174  /**********************************************************
175  * Calculate misc constants
176  *********************************************************/
177 
178  float_tt rx = 1.0/ax;
179  float_tt rx2 = rx * rx;
180  float_tt ry = 1.0/by;
181  float_tt ry2 = ry * ry;
182 
183  ixmid = m_nx/2;
184  iymid = m_ny/2;
185 
186  // Start in Fourier space
187  ToFourierSpace();
188 
189  /* convert convergence angle from mrad to rad */
190  alpha = 0.001*m_alpha;
191  k2max = sin(alpha)/m_wavlen; /* = K0*sin(alpha) */
192  k2max = k2max * k2max;
193 
194  /* Calculate MTF
195  NOTE zero freg is in the bottom left corner and
196  expandes into all other corners - not in the center
197  this is required for FFT
198 
199  PIXEL = diagonal width of pixel squared
200  if a pixel is on the apertur boundary give it a weight
201  of 1/2 otherwise 1 or 0
202  */
203  pixel = ( rx2 + ry2 );
204  scale = 1.0/sqrt((double)m_nx*(double)m_ny);
205 
206  for(iy=0; iy<m_ny; iy++) {
207  float_tt ky = (float_tt) iy;
208  if( iy > iymid ) ky = (double) (iy-m_ny);
209  float_tt ky2 = ky*ky*ry2;
210  for(ix=0; ix<m_nx; ix++) {
211  float_tt kx = (double) ix;
212  if( ix > ixmid ) kx = (double) (ix-m_nx);
213  float_tt k2 = kx*kx*rx2 + ky2;
214  float_tt ktheta2 = k2*(m_wavlen*m_wavlen);
215  float_tt ktheta = sqrt(ktheta2);
216  float_tt phi = atan2(ry*ky,rx*kx);
217 
218  // defocus, astigmatism, and shift:
219  float_tt chi = ktheta2*(m_df0+delta + m_astigMag*cos(2.0*(phi-m_astigAngle)))/2.0;
220  ktheta2 *= ktheta; // ktheta^3
221  if ((m_a33 > 0) || (m_a31 > 0)) {
222  chi += ktheta2*(m_a33*cos(3.0*(phi-m_phi33))+m_a31*cos(phi-m_phi31))/3.0;
223  }
224  ktheta2 *= ktheta; // ktheta^4
225  if ((m_a44 > 0) || (m_a42 > 0) || (m_Cs != 0)) {
226  chi += ktheta2*(m_a44*cos(4.0*(phi-m_phi44))+m_a42*cos(2.0*(phi-m_phi42))+m_Cs)/4.0;
227  }
228  ktheta2 *= ktheta; // ktheta^5
229  if ((m_a55 > 0) || (m_a53 > 0) || (m_a51 > 0)) {
230  chi += ktheta2*(m_a55*cos(5.0*(phi-m_phi55))+m_a53*cos(3.0*(phi-m_phi53))+m_a51*cos(phi-m_phi51))/5.0;
231  }
232  ktheta2 *= ktheta; // ktheta^6
233  if ((m_a66 > 0) || (m_a64 > 0) || (m_a62 = 0) || (m_C5 != 0)) {
234  chi += ktheta2*(m_a66*cos(6.0*(phi-m_phi66))+m_a64*cos(4.0*(phi-m_phi64))+m_a62*cos(2.0*(phi-m_phi62))+m_C5)/6.0;
235  }
236 
237  chi *= 2*M_PI/m_wavlen;
238  chi -= 2.0*M_PI*( (dx*kx/ax) + (dy*ky/by) );
239 
240  if ( ( m_ismoth != 0) &&
241  ( fabs(k2-k2max) <= pixel)) {
242  m_wave[ix+m_nx*iy][0]= (float) ( 0.5*scale * cos(chi));
243  m_wave[ix+m_nx*iy][1]= (float) (-0.5*scale* sin(chi));
244  }
245  else if ( k2 <= k2max ) {
246  m_wave[ix+m_nx*iy][0]= (float) scale * cos(chi);
247  m_wave[ix+m_nx*iy][1]= (float) -scale * sin(chi);
248  }
249  else {
250  m_wave[ix+m_nx*iy][0] = m_wave[ix+m_nx*iy][1] = 0.0f;
251  }
252  }
253  }
254  /* Fourier transform into real space */
255  ToRealSpace();
256  /**********************************************************
257  * display cross section of probe intensity
258  */
259 
260  /* multiply with gaussian in Real Space in order to avoid artifacts */
261  if (m_gaussFlag) {
262  for( ix=0; ix<m_nx; ix++) {
263  for( iy=0; iy<m_ny; iy++) {
264  r = exp(-((ix-m_nx/2)*(ix-m_nx/2)+(iy-m_ny/2)*(iy-m_ny/2))/(m_nx*m_nx*m_gaussScale));
265  m_wave[ix+m_nx*iy][0] *= (float)r;
266  m_wave[ix+m_nx*iy][1] *= (float)r;
267  }
268  }
269  }
270 
271  /* Apply AIS aperture in Real Space */
272  // printf("center: %g,%g\n",dx,dy);
273  if (m_aAIS > 0) {
274  for( ix=0; ix<m_nx; ix++) {
275  for( iy=0; iy<m_ny; iy++) {
276  x = ix*m_dx-dx;
277  y = iy*m_dy-dy;
278  r = sqrt(x*x+y*y);
279  delta = r-0.5*m_aAIS+edge;
280  if (delta > 0) {
281  m_wave[ix+m_nx*iy][0] = 0;
282  m_wave[ix+m_nx*iy][1] = 0;
283  }
284  else if (delta >= -edge) {
285  scale = 0.5*(1-cos(M_PI*delta/edge));
286  m_wave[ix+m_nx*iy][0] = scale*m_wave[ix+m_nx*iy][0];
287  m_wave[ix+m_nx*iy][1] = scale*m_wave[ix+m_nx*iy][1];
288  }
289  }
290  }
291  }
292 
293  /* Normalize probe intensity to unity */
294 
295  /*
296  float_tt sum = 0.0;
297  for( ix=0; ix<m_nx; ix++)
298  for( iy=0; iy<m_ny; iy++)
299  sum += m_wave[ix+m_nx*iy][0]*m_wave[ix+m_nx*iy][0]
300  + m_wave[ix+m_nx*iy][1]*m_wave[ix+m_nx*iy][1];
301 
302  scale = 1.0 / sum;
303  scale = scale * ((double)m_nx) * ((double)m_ny);
304  scale = (double) sqrt( scale );
305 
306  for( ix=0; ix<m_nx; ix++)
307  for( iy=0; iy<m_ny; iy++) {
308  m_wave[ix+m_nx*iy][0] *= (float) scale;
309  m_wave[ix+m_nx*iy][1] *= (float) scale;
310  }
311  */
312 
313  /* Output results and find min and max to echo
314  remember that complex pix are stored in the file in FORTRAN
315  order for compatability
316  */
317 
318  rmin = m_wave[0][0];
319  rmax = rmin;
320  aimin = m_wave[0][1];
321  aimax = aimin;
322  for( iy=0; iy<m_ny; iy++) {
323  for( ix=0; ix<m_nx; ix++) {
324  if( m_wave[ix+m_nx*iy][0] < rmin ) rmin = m_wave[ix+m_nx*iy][0];
325  if( m_wave[ix+m_nx*iy][0] > rmax ) rmax = m_wave[ix+m_nx*iy][0];
326  if( m_wave[ix+m_nx*iy][1] < aimin ) aimin = m_wave[ix+m_nx*iy][1];
327  if( m_wave[ix+m_nx*iy][1] > aimax ) aimax = m_wave[ix+m_nx*iy][1];
328  }
329  }
330  m_rmin = rmin;
331  m_rmax = rmax;
332  m_aimin = aimin;
333  m_aimax = aimax;
334 
335  /**********************************************************/
336 
337 } /* end probe() */
