232 {
233  /* Pointer used to create a new set of attributes
234  */
235  pcifParser_parse_SCOPE newAttributes;
236 
237  /* Allocate the memory for a new structure if we need one.
238  */
239  if (ctx->pcifParser_parseStack->size(ctx->pcifParser_parseStack) > ctx->pcifParser_parseStack_limit)
240  {
241  // The current limit value was less than the number of scopes available on the stack so
242  // we can just reuse one. Our limit tracks the stack count, so the index of the entry we want
243  // is one less than that, or conveniently, the current value of limit.
244  //
245  newAttributes = (pcifParser_parse_SCOPE)ctx->pcifParser_parseStack->get(ctx->pcifParser_parseStack, ctx->pcifParser_parseStack_limit);
246  }
247  else
248  {
249  // Need a new allocation
250  //
251  newAttributes = (pcifParser_parse_SCOPE) ANTLR3_MALLOC(sizeof(cifParser_parse_SCOPE));
252  if (newAttributes != NULL)
253  {
254  /* Standard ANTLR3 library implementation
255  */
256  ctx->pcifParser_parseStack->push(ctx->pcifParser_parseStack, newAttributes, (void (*)(void *))parseFree);
257  }
258  }

261  //
262  newAttributes->free = NULL;
263 
264  // Indicate the position in the available stack that the current level is at
265  //
266  ctx->pcifParser_parseStack_limit++;
267 
268  /* Return value is the pointer to the new entry, which may be used locally
269  * without de-referencing via the context.
270  */
271  return newAttributes;

517 {
518  // First see if the user defined a function they want to be called when a
519  // scope is popped/freed.
520  //

522  //

526  }
527 
528  // Now we decrement the scope's upper limit bound. We do not actually pop the scope as
529  // we want to reuse scope entries if we do continuous push and pops. Most scopes don't

532  ctx->pcifParser_parseStack_limit--;
533  SCOPE_TOP(parse) = (pcifParser_parse_SCOPE)(ctx->pcifParser_parseStack->get(ctx->pcifParser_parseStack, ctx->pcifParser_parseStack_limit - 1));
534 }
