42 void C3DPotential::AtomBoxLookUp(complex_tt &val, int Znum, float_tt x, float_tt y, float_tt z, float_tt B)
43 {
44  float_tt dx, dy, dz;
45  int ix, iy, iz;
46 
47  // does the atom box lookup or calculation
48  CPotential::AtomBoxLookUp(val, Znum, x, y, z, B);
49 
50  /***************************************************************
51  * Do the trilinear interpolation
52  */
53  val[0] = 0.0;
54  val[1] = 0.0;
55  if (x*x+y*y+z*z > m_atomRadius2) {
56  return;
57  }
58  x = fabs(x);
59  y = fabs(y);
60  z = fabs(z);
61  ix = (int)(x/m_ddx);
62  iy = (int)(y/m_ddy);
63  iz = (int)(z/m_ddz);
64  dx = x-(float_tt)ix*m_ddx;
65  dy = y-(float_tt)iy*m_ddy;
66  dz = z-(float_tt)iz*m_ddz;
67  if ((dx < 0) || (dy<0) || (dz<0)) {
68  /* printf("Warning, dx(%g), dy(%g), dz(%g) < 0, (x=%g, y=%g, z=%g)\n",dx,dy,dz,x,y,z);
69  */
70  if (dx < 0) dx = 0.0;
71  if (dy < 0) dy = 0.0;
72  if (dz < 0) dz = 0.0;
73  }
74 
75  if (m_atomBoxes[Znum]->B > 0) {
76  val[0] = (1.0-dz)*((1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[iz][ix][iy][0]+
77  dx*m_atomBoxes[Znum]->potential[iz][ix+1][iy][0])+
78  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[iz][ix][iy+1][0]+
79  dx*m_atomBoxes[Znum]->potential[iz][ix+1][iy+1][0]))+
80  dz*((1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[iz+1][ix][iy][0]+
81  dx*m_atomBoxes[Znum]->potential[iz+1][ix+1][iy][0])+
82  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[iz+1][ix][iy+1][0]+
83  dx*m_atomBoxes[Znum]->potential[iz+1][ix+1][iy+1][0]));
84  val[1] = (1.0-dz)*((1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[iz][ix][iy][1]+
85  dx*m_atomBoxes[Znum]->potential[iz][ix+1][iy][1])+
86  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[iz][ix][iy+1][1]+
87  dx*m_atomBoxes[Znum]->potential[iz][ix+1][iy+1][1]))+
88  dz*((1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[iz+1][ix][iy][1]+
89  dx*m_atomBoxes[Znum]->potential[iz+1][ix+1][iy][1])+
90  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[iz+1][ix][iy+1][1]+
91  dx*m_atomBoxes[Znum]->potential[iz+1][ix+1][iy+1][1]));
92  }
93  else {
94  val[0] = (1.0-dz)*((1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->rpotential[iz][ix][iy]+
95  dx*m_atomBoxes[Znum]->rpotential[iz][ix+1][iy])+
96  dy*((1.0-dx)*m_atomBoxes[Znum]->rpotential[iz][ix][iy+1]+
97  dx*m_atomBoxes[Znum]->rpotential[iz][ix+1][iy+1]))+

100  dy*((1.0-dx)*m_atomBoxes[Znum]->rpotential[iz+1][ix][iy+1]+
101  dx*m_atomBoxes[Znum]->rpotential[iz+1][ix+1][iy+1]));
102  }

105 bool C3DPotential::CheckAtomZInBounds(float_tt atomZ)
106 {
107  /*
108  * c = the thickness of the current slab.
109  *
110  * if the z-position of this atom is outside the potential slab
111  * we won't consider it and skip to the next
112  */
113  return ((atomZ - m_atomRadius > m_c) && (atomZ + m_atomRadius + m_sliceThickness >= 0));

125 void C3DPotential::_AddAtomRealSpace(std::vector<atom>::iterator &atom,
126  float_tt atomBoxX, unsigned ix,
127  float_tt atomBoxY, unsigned iy,
128  float_tt atomBoxZ, unsigned iAtomZ)
129 {
130  unsigned iz;
131  complex_tt dPot;
132 
133  /* calculate the range which we have left to cover with z-variation */
134  /* iRadZ is the number of slices (rounded up) that this atom
135  * will contribute to, given its current x,y-radius
136  */
137  float_tt r2sqr = atomBoxX*atomBoxX + atomBoxY*atomBoxY;
138  // TODO: iRadZ is also calculated in the base class, one level up. Which is correct?
139  unsigned iRadZ = (unsigned)(sqrt(m_atomRadius2-r2sqr)/m_cz[0]+1.0);
140  /* loop through the slices that this atoms contributes to */
141  for (int iaz=-m_iRadZ;iaz <=m_iRadZ;iaz++) {
142  if (!m_periodicZ) {
143  if (iaz+iAtomZ < 0) {
144  if (-iAtomZ <= m_iRadZ) iaz = -iAtomZ;
145  else break;
146  if (abs(iaz)>m_nslices) break;
147  }
148  if (iaz+iAtomZ >= m_nslices) break;
149  }
150  atomBoxZ = (double)(iAtomZ+iaz+0.5)*m_cz[0]-atomBoxZ;
151  /* shift into the positive range */
152  iz = (iaz+iAtomZ+32*m_nslices) % m_nslices;
153  /* x,y,z is the true vector from the atom center
154  * We can look up the proj potential at that spot
155  * using trilinear extrapolation.
156  */
157  AtomBoxLookUp(dPot,atom->Znum,atomBoxX,atomBoxY,atomBoxZ,
158  m_tds ? 0 : atom->dw);
159  // printf("access: %d %d %d\n",iz,ix,iy);
160  unsigned idx=ix*m_ny+iy;
161  m_trans[iz][idx]+=dPot;
162  //m_trans[iz][ix][iy][0] += dPot[0];
163  //m_trans[iz][ix][iy][1] += dPot[1];
164  } /* end of for iaz=-iRadZ .. iRadZ */
165 }
