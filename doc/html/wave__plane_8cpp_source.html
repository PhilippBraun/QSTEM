70 void CPlaneWave::TiltBeam(bool tiltBack)
71 {
72  if ((m_btiltx != 0) || (m_btilty != 0))
73  {
74 
75  int direction = tiltBack ? -1 : 1;
76 
77  // produce a tilted wave function (btiltx,btilty):
78  float_tt ktx = direction*2.0*M_PI*sin(m_btiltx)/GetWavelength();
79  float_tt kty = direction*2.0*M_PI*sin(m_btilty)/GetWavelength();
80  unsigned px=m_nx*m_ny;
81  for (unsigned i=0; i<px; i++)
82  {
83  unsigned ix=i%m_nx;
84  unsigned iy=i/m_ny;
85  float_tt x = m_dx*(ix-m_nx/2);
86  float_tt y = m_dy*(iy-m_ny/2);
87  m_wave[i][0] = (float)cos(ktx*x+kty*y);
88  m_wave[i][1] = (float)sin(ktx*x+kty*y);
89  }
90  }
91 }
