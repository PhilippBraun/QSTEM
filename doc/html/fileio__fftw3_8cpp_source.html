270  if (superCell.atoms[i].Znum == idArray[id]) break;
271  }
272  if (id>idArrayPtr) {

284  /**************************************************************************************
285  * Now that we found all the species present in the amorphous phase we can go ahead and
286  * make a list of them, together with the number of atoms present of each kind
287  */
288  for (id=0;id<=idArrayPtr;id++) {
289  if (idCountArray[id] > 0) {
290  if (idArray[id] >= MAX_MASS_INDEX) {
291  printf("mass exceeds array!!! - extend array!\n");

295  case 14: charge = 4.0; break;
296  case 7: charge = -3.0; break;

298  case 39: charge = 3.0; break;

307  }
308  }

311  * The list of the potential parameters is next
312  */

315  for (id=0;id<=idArrayPtr;id++)
316  for (j=id;j<=idArrayPtr;j++) {

318  switch (10*(id+1) + (j+1)) {
319  case 11: b = 0.0; break; // N -N
320  case 12: b = 4.5; break; // N -Si
321  case 13: b = 4.5; break; // N -Y
322  case 14: b = 0.0; break; // N -O
323  case 22: b = 1.8770; break; // Si-Si
324  case 23: b = 4.4200; break; // Si-Y
325  case 24: b = 2.9620; break; // Si-O
326  case 33: b = 9.9706; break; // Y -Y
327  case 34: b = 6.0802; break; // Y -O
328  case 44: b = 0.7254; break; // O -O
329  default : b=0.0;
330  }
331  b = b*6.022e4; // *1e-16*6.022e23*1e-3
332  fprintf(fp,"%d %d %g %g %g\n",id+1,j+1,A,b,C);
333 
334  }
335  fprintf(fp,"end\n");

337  /*************************************************************************
338  * We can now write down the size of the MD cell:
339  */
340  fprintf(fp,"%g %g %g 90 90 90 1 1 1\n",superCell.ax,superCell.by,superCell.cz);
341 
342  /****************************************************************************
343  * now list the position of each one of the atoms in the amorphous phase
344  * in FRACTIONAL COORDINATES !!!
345  */
346  for(i=nstart;i<nstop;i++) {
347  x = superCell.atoms[i].x/superCell.ax;
348  y = superCell.atoms[i].y/superCell.by;
349  z = superCell.atoms[i].z/superCell.cz;

354  elTable[2*superCell.atoms[i].Znum-1],x,y,z);
355  }
356  if (nstart > 0)
357  fprintf(fp,"frame1 %7.5f %7.5f %7.5f 1 0 0 0\n",superCell.cmx,superCell.cmy,superCell.cmz);
358  fprintf(fp,"end\n");
359  } // end of if fp != NULL
360 }
