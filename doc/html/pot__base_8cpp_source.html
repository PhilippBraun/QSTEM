49 void CPotential::Initialize()
50 {
51  m_ddx = m_dx/(double)OVERSAMPLING;
52  m_ddy = m_dy/(double)OVERSAMPLING;
53  m_ddz = m_dz/(double)OVERSAMPLINGZ;
54 
55  /* For now we don't care, if the box has only small
56  * prime factors, because we will not fourier transform it
57  * especially not very often.
58  */
59  m_boxNx = (int)(m_atomRadius/m_ddx+2.0);
60  m_boxNy = (int)(m_atomRadius/m_ddy+2.0);
61 
62  m_c = m_sliceThickness * m_nslices;
63  m_dr = m_dx/OVERSAMPLING; // define step width in which radial V(r,z) is defined
64  m_iRadX = (int)ceil(m_atomRadius/m_dx);
65  m_iRadY = (int)ceil(m_atomRadius/m_dy);
66  m_iRadZ = (int)ceil(m_atomRadius/m_sliceThickness);
67  m_iRad2 = m_iRadX*m_iRadX+m_iRadY*m_iRadY;
68  m_atomRadius2 = m_atomRadius * m_atomRadius;
69 
70  m_cz.resize(m_nslices);
71  m_slicePos.resize(m_nslices);
72 
73  if (m_sliceThickness == 0)
74  m_cz[0] = m_c/(float_tt)m_nslices;
75  else
76  m_cz[0] = m_sliceThickness;
77 
78  m_slicePos[0] = m_zOffset;

122  int numRead = 0,dummy;
123 
124  /* initialize all the atoms to non-used */
125  if (!m_atomBoxes.count(Znum)) {
126  m_atomBoxes[Znum]=atomBoxPtr(new atomBox());
127  m_atomBoxes[Znum]->potential = NULL;
128  m_atomBoxes[Znum]->rpotential = NULL;

130 
131  if (m_printLevel > 2)
132  printf("Atombox has real space resolution of %g x %g x %gA (%d x %d x %d pixels)\n",

134  }
135  // printf("Debugging: %d %g %g: %g\n",Znum,m_atomBoxes[Znum]->B,B,fabs(m_atomBoxes[Znum]->B - B));
136 
137  /* Creating/Reading a atombox for every new kind of atom, but only as needed */
138  if (fabs(m_atomBoxes[Znum]->B - B) > 1e-6) {
139  // printf("Debugging 1 (%d: %.7g-%.7g= %.7g), %d\n",
140  // Znum,m_atomBoxes[Znum]->B,B,fabs(m_atomBoxes[Znum]->B - B),fabs(m_atomBoxes[Znum]->B - B) > 1e-6);
141  m_atomBoxes[Znum]->B = B;
142  /* Open the file with the projected potential for this particular element
143  */
144  sprintf(fileName,"potential_%d_B%d.prj",Znum,(int)(100.0*B));
145  if ((fp=fopen(fileName,"r")) == NULL) {
146  sprintf(systStr,"scatpot %s %d %g %d %d %d %g %g %g %d %g",
147  fileName,Znum,B,boxNx,boxNy,boxNz,ddx,ddy,ddz,OVERSAMPLINGZ,m_v0);
148  if (m_printLevel > 2) {
149  printf("Could not find precalculated potential for Z=%d,"
150  " will calculate now.\n",Znum);
151  printf("Calling: %s\n",systStr);
152  }
153  system(systStr);
154  for (dummy=0;dummy < 10000;dummy++);
155  if ((fp=fopen(fileName,"r")) == NULL) {
156  printf("cannot calculate projected potential using scatpot - exit!\n");
157  exit(0);
158  }
159 
160  }
161  fgets(systStr,250,fp);
162  sscanf(systStr,"%d %f %d %d %d %f %f %f %d %f\n",
163  &tZ, &tB, &tnx, &tny, &tnz, &tdx, &tdy, &tdz, &tzOversample, &tv0);
164  /* If the parameters in the file don't match the current ones,
165  * we need to create a new potential file
166  */
167  if ((tZ != Znum) || (fabs(tB-B)>1e-6) || (tnx != boxNx) || (tny != boxNy) || (tnz != boxNz) ||
168  (fabs(tdx-ddx) > 1e-5) || (fabs(tdy-ddy) > 1e-5) || (fabs(tdz-ddz) > 1e-5) ||

171  printf("Potential input file %s has the wrong parameters\n",fileName);

179  /* Close the old file, Create a new potential file now
180  */
181  fclose(fp);
182  sprintf(systStr,"scatpot %s %d %g %d %d %d %g %g %g %d %g",
183  fileName,Znum,B,boxNx,boxNy,boxNz,ddx,ddy,ddz,OVERSAMPLINGZ,m_v0);
184  system(systStr);
185  if ((fp=fopen(fileName,"r")) == NULL) {
186  printf("cannot calculate projected potential using scatpot - exit!\n");
187  exit(0);
188  }
189  fgets(systStr,250,fp);
190  }
191 
192  /* Finally we can read in the projected potential
193  */
194  if (B == 0) {
195  m_atomBoxes[Znum]->rpotential = float3D(boxNz,boxNx,boxNy,"atomBox");
196  numRead = fread(m_atomBoxes[Znum]->rpotential[0][0],sizeof(float_tt),
197  (size_t)(boxNx*boxNy*boxNz),fp);
198  }
199  else {
200  m_atomBoxes[Znum]->potential = complex3D(boxNz,boxNx,boxNy,"atomBox");
201  numRead = fread(m_atomBoxes[Znum]->potential[0][0],sizeof(complex_tt),

203  }
204 
205  /* writeImage_old(m_atomBoxes[Znum]->potential[0],boxNx,boxNy, 0.0,"potential.img");
206  system("showimage potential.img");
207  */
208  fclose(fp);
209 
210  if (numRead == boxNx*boxNy*boxNz) {
211  if (m_printLevel > 1)
212  printf("Sucessfully read in the projected potential\n");
213  }
214  else {
215  printf("error while reading potential file %s: read %d of %d values\n",
216  fileName,numRead,boxNx*boxNy*boxNz);
217  exit(0);
218  }

254 void CPotential::SliceSetup()
255 {
256  FILE *sliceFp;
257  char buf[BUF_LEN];
258  /**************************************************************
259  * setup the slices with their start and end positions
260  * then loop through all the atoms and add their potential to
261  * the slice that their potential reaches into (up to RMAX)
262  *************************************************************/
263 
264  for (unsigned i=1;i<m_nslices;i++) {
265  if (sliceFp == NULL) m_cz[i] = m_cz[0];
266  /* need to all be the same for fast 3D-FFT method, otherwise OK to be different */
267  else {
268  fgets(buf,BUF_LEN,sliceFp);
269  m_cz[i] = atof(buf);
270  }
271  // Slice position is last slice position + half width of last slice + half width of this slice
272  m_slicePos[i] = m_slicePos[i-1]+m_cz[i-1]/2.0+m_cz[i]/2.0;
273  }
274  m_trans.resize(m_nslices);
275  // If we are going to read the potential, then we need to size the slices according to the first read pot slice
276  if (m_readPotential)
277  {
278  }
279  // If we are going to calculate the potential, then we need to size the slices according to the size of the
280  // structure and the corresponding resolution.
281  else
282  {
283  float_tt max_x, min_x, max_y, min_y;
284  m_crystal->GetCrystalBoundaries(min_x, max_x, min_y, max_y);
285  m_nx = ceil((max_x-min_x)/m_dx);
286  m_ny = ceil((max_y-min_y)/m_dy);
287  }
288  ResizeSlices();
289 }

298 void CPotential::CenterAtomZ(std::vector<atom>::iterator &atom, float_tt &z)
299 {
300 
301  /*
302  * Since cellDiv is always >=1, and divCount starts at 0, the actual position
303  * of this atom with the super cell is given by:
304  */
305  /* c*(float_tt)((*muls).cellDiv-divCount-1) will pick the right super-cell
306  * division in the big super-cell
307  * The z-offset 0.5*cz[0] will position atoms at z=0 into the middle of the first
308  * slice.
309  */
310  z = atom->z-m_c*(float_tt)(m_cellDiv-m_divCount-1) + m_zOffset
311  -(0.5*m_sliceThickness*(1-(int)m_centerSlices));
312 }

412 void CPotential::AddAtomRealSpace(std::vector<atom>::iterator &atom,
413  float_tt atomX, float_tt atomY, float_tt atomZ)
414 {
415  unsigned iatom = atom-m_atoms.begin();
416 
417  CenterAtomZ(atom, atomZ);
418 
419  /* Warning: will assume constant slice thickness ! */
420  /* do not round here: atomX=0..dx -> iAtomX=0 */
421  unsigned iAtomX = (int)floor(atomX/m_dx);
422  unsigned iAtomY = (int)floor(atomY/m_dy);
423  unsigned iAtomZ = (int)floor(atomZ/m_cz[0]);
424 
425  if (m_displayPotCalcInterval > 0) {
426  if ((m_printLevel>=3) && ((iatom+1) % m_displayPotCalcInterval == 0)) {
427  printf("adding atom %d [%.3f %.3f %.3f (%.3f)], Z=%d\n",
428  iatom+1,atomX+m_offsetX,atomY+m_offsetY,
429  atom->z,atomZ,atom->Znum);
430  }
431  }
432 
433  for (int iax = -m_iRadX;iax<=m_iRadX;iax++) {
434  if (!m_periodicXY) {
435  if (iax+iAtomX < 0) {
436  iax = -iAtomX;
437  if (abs(iax)>m_iRadX) break;
438  }
439  if (iax+iAtomX >= m_nx) break;
440  }
441  float_tt x = (iAtomX+iax)*m_dx-atomX;
442  unsigned ix = (iax+iAtomX+16*m_nx) % m_nx; /* shift into the positive range */
443  for (int iay=-m_iRadY;iay<=m_iRadY;iay++) {
444  if (!m_periodicXY) {
445  if (iay+iAtomY < 0) {
446  iay = -iAtomY;
447  if (abs(iay)>m_iRadY) break;
448  }
449  if (iay+iAtomY >= m_ny) break;
450  }
451  float_tt y = (iAtomY+iay)*m_dy-atomY;
452  unsigned iy = (iay+iAtomY+16*m_ny) % m_ny; /* shift into the positive range */
453  float_tt r2sqr = x*x + y*y;
454  if (r2sqr <= m_atomRadius2)
455  {
456  // This (virtual) method is meant to be implemented by subclasses,
457  // for specific functionality varying by dimensionality.
458  _AddAtomRealSpace(atom, x, ix, y, iy, atomZ, iAtomZ);
459  }
460  }
461  }

540  if (sf < 0) return 0.0;
541  return(sf);

546 */

549 /*------------------ splinh() -----------------------------*/
550 /*
551  fit a quasi-Hermite cubic spline
552 
553  [1] Spline fit as in H.Akima, J. ACM 17(1970)p.589-602
554  'A New Method of Interpolation and Smooth
555  Curve Fitting Based on Local Procedures'
556 
557  [2] H.Akima, Comm. ACM, 15(1972)p.914-918
558 
559  E. Kirkland 4-JUL-85
560  changed zero test to be a small nonzero number 8-jul-85 ejk
561  converted to C 24-jun-1995 ejk
562 
563  The inputs are:
564  x[n] = array of x values in ascending order, each X(I) must
565  be unique
566  y[n] = array of y values corresponding to X(N)
567  n = number of data points must be 2 or greater
568 
569  The outputs are (with z=x-x(i)):
570  b[n] = array of spline coeficients for (x-x[i])
571  c[n] = array of spline coeficients for (x-x[i])**2
572  d[n] = array of spline coeficients for (x-x[i])**3
573  ( x[i] <= x <= x[i+1] )
574  To interpolate y(x) = yi + bi*z + c*z*z + d*z*z*z
575 
576  The coeficients b[i], c[i], d[i] refer to the x[i] to x[i+1]
577  interval. NOTE that the last set of coefficients,
578  b[n-1], c[n-1], d[n-1] are meaningless.
579 */
580 void CPotential::splinh( float_tt x[], float_tt y[],
581  float_tt b[], float_tt c[], float_tt d[], int n)
582 {
583 #define SMALL 1.0e-25
584 
585  int i, nm1, nm4;
586  float_tt m1, m2, m3, m4, m5, t1, t2, m54, m43, m32, m21, x43;
587 
588  if( n < 4) return;
589 
590  /* Do the first end point (special case),
591 and get starting values */
592 
593  m5 = ( y[3] - y[2] ) / ( x[3] - x[2] ); /* mx = slope at pt x */
594  m4 = ( y[2] - y[1] ) / ( x[2] - x[1] );
595  m3 = ( y[1] - y[0] ) / ( x[1] - x[0] );
596 
597  m2 = m3 + m3 - m4; /* eq. (9) of reference [1] */
598  m1 = m2 + m2 - m3;
599 
600  m54 = fabs( m5 - m4);
601  m43 = fabs( m4 - m3);
602  m32 = fabs( m3 - m2);
603  m21 = fabs( m2 - m1);
604 
605  if ( (m43+m21) > SMALL )
606  t1 = ( m43*m2 + m21*m3 ) / ( m43 + m21 );
607  else
608  t1 = 0.5 * ( m2 + m3 );
609 
610  /* Do everything up to the last end points */
611 
612  nm1 = n-1;
613  nm4 = n-4;
614 
615  for( i=0; i<nm1; i++) {
616 
617  if( (m54+m32) > SMALL )
618  t2= (m54*m3 + m32*m4) / (m54 + m32);
619  else
620  t2 = 0.5* ( m3 + m4 );
621 
622  x43 = x[i+1] - x[i];
623  b[i] = t1;
624  c[i] = ( 3.0*m3 - t1 - t1 - t2 ) /x43;
625  d[i] = ( t1 + t2 - m3 - m3 ) / ( x43*x43 );
626 
627  m1 = m2;
628  m2 = m3;
629  m3 = m4;
630  m4 = m5;
631  if( i < nm4 ) {
632  m5 = ( y[i+4] - y[i+3] ) / ( x[i+4] - x[i+3] );
633  } else {
634  m5 = m4 + m4 - m3;
635  }
636 
637  m21 = m32;
638  m32 = m43;
639  m43 = m54;
640  m54 = fabs( m5 - m4 );
641  t1 = t2;
642  }
643 
644  return;
645 
646 } /* end splinh() */
647 
648 /*----------------------- seval() ----------------------*/
649 /*
650  Interpolate from cubic spline coefficients
651 
652  E. Kirkland 4-JUL-85
653  modified to do a binary search for efficiency 13-Oct-1994 ejk
654  converted to C 26-jun-1995 ejk
655  fixed problem on end-of-range 16-July-1995 ejk
656 
657  The inputs are:
658  x[n] = array of x values in ascending order, each x[i] must
659  be unique
660  y[n] = array of y values corresponding to x[n]
661  b[n] = array of spline coeficients for (x-x[i])
662  c[n] = array of spline coeficients for (x-x[i])**2
663  d[n] = array of spline coeficients for (x-x[i])**3
664  n = number of data points
665  x0 = the x value to interpolate at
666  (x[i] <= x <= x[i+1]) and all inputs remain unchanged
667 
668  The value returned is the interpolated y value.
669 
670  The coeficients b[i], c[i], d[i] refer to the x[i] to x[i+1]
671  interval. NOTE that the last set of coefficients,
672  b[n-1], c[n-1], d[n-1] are meaningless.
673 */
674 float_tt CPotential::seval( float_tt *x, float_tt *y, float_tt *b, float_tt *c,
675  float_tt *d, int n, float_tt x0 )
676 {
677  int i, j, k;
678  float_tt z, seval1;
679 
680  /* exit if x0 is outside the spline range */
681  if( x0 <= x[0] ) i = 0;
682  else if( x0 >= x[n-2] ) i = n-2;
683  else {
684  i = 0;
685  j = n;
686  do{ k = ( i + j ) / 2 ;
687  if( x0 < x[k] ) j = k;
688  else if( x0 >= x[k] ) i = k;
689  } while ( (j-i) > 1 );
690  }
691 
692  z = x0 - x[i];
693  seval1 = y[i] + ( b[i] + ( c[i] + d[i] *z ) *z) *z;
694 
695  return( seval1 );
696 
697 } /* end seval() */
