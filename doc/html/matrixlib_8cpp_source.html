47 void ludcmp(float_tt **a, int n, int *indx, float_tt *d) {
48  /* Given a matrix a[1..n][1..n], this routine replaces it by the
49  LU decomposition of a rowwise permutation of itself.
50  a and n are input. a is output, arranged as in equation (2.3.14) above;
51  indx[1..n] is an output vector that records the row permutation e ected
52  by the partial pivoting;
53  d is output as 1 depending on whether the number of row interchanges
54  was even or odd, respectively. This routine is used in combination with
55  lubksb to solve linear equations or invert a matrix.
56  */
57  int i,imax=0,j,k;
58  float_tt big,dum,sum,temp;
59  /*
60  vv stores the implicit scaling of each row.
61  */
62  std::vector<float_tt> vv(n);
63 
64  *d=1.0; //No row interchanges yet.
65  for (i=0;i<n;i++) {
66  // Loop over rows to get the implicit scaling information.
67  big=0.0;
68  for (j=0;j<n;j++)
69  if ((temp=fabs(a[i][j])) > big)
70  big=temp;
71  if (big == 0.0) {
72  printf("Singular matrix in routine ludcmp\n");
73  exit(0);
74  }
75  //No nonzero largest element.
76  vv[i]=1.0/big; // Save the scaling.
77  }
78  for (j=0;j<n;j++) {
79  // This is the loop over columns of Crout's method.
80  for (i=0;i<j;i++) { //This is equation (2.3.12) except for i = j.
81  sum=a[i][j];
82  for (k=0;k<i;k++)
83  sum -= a[i][k]*a[k][j];
84  a[i][j]=sum;
85  }
86  big=0.0;
87  //Initialize for the search for largest pivot element.
88  for (i=j;i<n;i++) {
89  //This is i = j of equation (2.3.12) and i = j +1: ::N of equation (2.3.13).
90  sum=a[i][j];
91  for (k=0;k<j;k++)
92  sum -= a[i][k]*a[k][j];
93  a[i][j]=sum;
94  if ( (dum=vv[i]*fabs(sum)) >= big) {
95  //Is the gure of merit for the pivot better than the best so far?
96  big=dum; imax=i;
97  }
98  }
99  if (j != imax) {
100  //Do we need to interchange rows?
101  for (k=0;k<n;k++) {
102  //Yes, do so...
103  dum=a[imax][k];
104  a[imax][k]=a[j][k];
105  a[j][k]=dum;
106  }
107  *d = -(*d); //...and change the parity of d.
108  vv[imax]=vv[j];
109  // Also interchange the scale factor.
110  }
111  indx[j]=imax;
112  if (a[j][j] == 0.0)
113  a[j][j]=TINY;
114  /*If the pivot element is zero the matrix is singular
115  (at least to the precision of the algorithm).
116  For some applications on singular matrices, it is
117  desirable to substitute TINY for zero.
118  */
119  if (j != n-1) { // Now, nally, divide by the pivot element.
120  dum=1.0/(a[j][j]);
121  for (i=j+1;i<n;i++) a[i][j] *= dum;
122  }
123  }
124  // Go back for the next column in the reduction.
125  //free(vv);
126  // free_vector(vv,1,n);
127 }

129 void lubksb(float_tt **a, int n, int *indx, float_tt b[]) {
130  /*
131  Solves the set of n linear equations A X = B.
132  Herea[0..n-1][0..n-1] is input, not as the matrix A but rather
133  as its LU decomposition, determined by the routine ludcmp.
134  indx[0..n-1] is input as the permutation vector returned by ludcmp.
135  b[0..n-1] is input as the right-hand side vector B, and returns
136  with the solution vector X. a, n, andindx are not modi ed by
137  this routine and can be left in place for successive calls with
138  di erent right-hand sides b. This routine takes into account
139  the possibility that b will begin with many zero elements,
140  so it is e cient for use in matrix inversion.
141  */
142  int i,ii=-1,ip,j;
143  float_tt sum;
144  for (i=0;i<n;i++) {
145  /* When ii is set to a positive value, it will become the index
146  of the rst nonvanishing element of b. Wenow do the forward
147  substitution, equation (2.3.6). The only new wrinkle is to
148  unscramble the permutation as we go.
149  */
150  ip=indx[i];
151  sum=b[ip];
152  b[ip]=b[i];
153  if (ii+1)
154  for (j=ii;j<=i-1;j++)
155  sum -= a[i][j]*b[j];
156  else
157  if (sum) ii=i;
158  /*
159  A nonzero element was encountered, so from now on we will have
160  to do the sums in the loop above.
161  */
162  b[i]=sum;
163  }
164  for (i=n-1;i>=0;i--) { //Now we do the backsubstitution, equation (2.3.7).
165  sum=b[i];
166  for (j=i+1;j<n;j++)
167  sum -= a[i][j]*b[j];
168  b[i]=sum/a[i][i]; // Store a component of the solution vector X.
169  }
170  //All done!
171 }

195  Output: returns the determinant of 'm'

197 float_tt det_3x3 (const float_tt *mat)
198 {
199  float_tt det;
200 
201  det = mat[0] * (mat[4] * mat[8] - mat[7] * mat[5])
202  - mat[1] * (mat[3] * mat[8] - mat[6] * mat[5])
203  + mat[2] * (mat[3] * mat[7] - mat[6] * mat[4]);
204 
205  return det;

285 void svdcmp1(float_tt **a, int m, int n, float_tt w[], float_tt **v) {
286  /* Given a matrix a[1..m][1..n], this routine computes its singular value decomposition,
287  A = U W V T .
288  The matrix U replaces a on output. The diagonal matrix of singular values
289  W is out- put as a vector w[1..n]. The matrix V (not the transpose V T ) is output
290  as v[1..n][1..n].
291  */
292  // uses: float_tt pythag(float_tt a, float_tt b);
293  int flag,i,its,j,jj,k,l=0,nm=0;
294  float_tt anorm,c,f,g,h,s,scale,x,y,z,*rv1;
295 
296  // rv1=vector(1,n); we will just alloacte memory for n+1 elements
297  rv1 = (float_tt *)malloc((n+1)*sizeof(float_tt));
298  g=scale=anorm=0.0; // Householder reduction to bidiagonal form.
299  for (i=1;i<=n;i++) {
300  l=i+1;
301  rv1[i]=scale*g;
302  g=s=scale=0.0;
303  if (i <= m) {
304  for (k=i;k<=m;k++)
305  scale += fabs(a[k][i]);
306  if (scale) {
307  for (k=i;k<=m;k++) {
308  a[k][i] /= scale;
309  s += a[k][i]*a[k][i];
310  }
311  f=a[i][i];
312  g = -SIGN(sqrt(s),f);
313  h=f*g-s;
314  a[i][i]=f-g;
315  for (j=l;j<=n;j++) {
316  for (s=0.0,k=i;k<=m;k++) s += a[k][i]*a[k][j];
317  f=s/h;
318  for (k=i;k<=m;k++) a[k][j] += f*a[k][i];
319  }
320  for (k=i;k<=m;k++) a[k][i] *= scale;
321  }
322  }
323  w[i]=scale *g;
324  g=s=scale=0.0;
325  if(i<= m && i != n) {
326  for (k=l;k<=n;k++) scale += fabs(a[i][k]);
327  if (scale) {
328 
329  for (k=l;k<=n;k++) {
330  a[i][k] /= scale;
331  s += a[i][k]*a[i][k];
332  }
333  f=a[i][l];
334  g = -SIGN(sqrt(s),f);
335  h=f*g-s;
336  a[i][l]=f-g;
337  for (k=l;k<=n;k++) rv1[k]=a[i][k]/h;
338  for (j=l;j<=m;j++) {
339  for (s=0.0,k=l;k<=n;k++)
340  s += a[j][k]*a[i][k];
341  for (k=l;k<=n;k++)
342  a[j][k] += s*rv1[k];
343  }
344  for (k=l;k<=n;k++) a[i][k] *= scale;
345  }
346  }
347  anorm=FMAX(anorm,(fabs(w[i])+fabs(rv1[i])));
348  }
349  for (i=n;i>=1;i--) { // Accumulation of right-hand transformations.
350  if(i < n) {
351  if (g) {
352  for (j=l;j<=n;j++) // Float_Tt division to avoid possible under ow.
353  v[j][i]=(a[i][j]/a[i][l])/g;
354  for (j=l;j<=n;j++) {
355  for (s=0.0,k=l;k<=n;k++) s += a[i][k]*v[k][j];
356  for (k=l;k<=n;k++) v[k][j] += s*v[k][i];
357  }
358  }
359  for (j=l;j<=n;j++) v[i][j]=v[j][i]=0.0;
360  }
361  v[i][i]=1.0;
362  g=rv1[i];
363  l=i;
364  }
365  for (i=IMIN(m,n);i>=1;i--) { // Accumulation of left-hand transformations.
366  l=i+1;
367  g=w[i];
368  for (j=l;j<=n;j++) a[i][j]=0.0;
369  if (g) {
370  g=1.0/g;
371  for (j=l;j<=n;j++) {
372  for (s=0.0,k=l;k<=m;k++) s += a[k][i]*a[k][j];
373  f=(s/a[i][i])*g;
374  for (k=i;k<=m;k++) a[k][j] += f*a[k][i];
375  }
376  for (j=i;j<=m;j++) a[j][i] *= g;
377  }
378  else for (j=i;j<=m;j++) a[j][i]=0.0; ++a[i][i];
379  }
380  for (k=n;k>=1;k--) {
381  // Diagonalization of the bidiagonal form: Loop over singular values,
382  // and over allowed iterations.
383  for (its=1;its<=30;its++) {
384  flag=1;
385  for (l=k;l>=1;l--) { // Test for splitting.
386  nm=l-1; // Note that rv1[1] is always zero.
387  if ((fabs(rv1[l])+anorm) == anorm) {
388  flag=0;
389  break;
390  }
391  if ((fabs(w[nm])+anorm) == anorm) break;
392  }
393  if (flag) {
394  c=0.0; // Cancellation of rv1[l], if l>1.
395  s=1.0;
396  for (i=l;i<=k;i++) {
397  f=s*rv1[i];
398  rv1[i]=c*rv1[i];
399  if ((fabs(f)+anorm) == anorm)
400  break;
401  g=w[i];
402  h=pythag(f,g);
403  w[i]=h;
404  h=1.0/h;
405  c=g*h;
406  s = -f*h;
407  for (j=1;j<=m;j++) {
408  y=a[j][nm];
409  z=a[j][i];
410  a[j][nm]=y*c+z*s;
411  a[j][i]=z*c-y*s;
412  }
413  }
414  }
415  z=w[k];
416  if (l == k) { //Convergence.
417  if (z < 0.0) { // Singular value is made nonnegative.
418  w[k] = -z;
419  for (j=1;j<=n;j++) v[j][k] = -v[j][k];
420  }
421  break;
422  }
423  if (its == 30) {
424  printf("no convergence in 30 svdcmp iterations");
425  exit(0);
426  }
427  x=w[l]; // Shiftfrom bottom 2-by-2minor.
428  nm=k-1;
429  y=w[nm];
430  g=rv1[nm];
431  h=rv1[k];
432  f=((y-z)*(y+z)+(g-h)*(g+h))/(2.0*h*y);
433  g=pythag(f,1.0);
434  f=((x-z)*(x+z)+h*((y/(f+SIGN(g,f)))-h))/x;
435  c=s=1.0;
436  // Next QR transformation:
437  for (j=l;j<=nm;j++) {
438  i=j+1;
439  g=rv1[i];
440  y=w[i];
441  h=s*g;
442  g=c*g;
443  z=pythag(f,h);
444  rv1[j]=z;
445  c=f/z;
446  s=h/z;
447  f=x*c+g*s;
448  g = g*c-x*s;
449  h=y*s;
450  y *= c;
451  for (jj=1;jj<=n;jj++) {
452  x=v[jj][j];
453  z=v[jj][i];
454  v[jj][j]=x*c+z*s;
455  v[jj][i]=z*c-x*s;
456  }
457  z=pythag(f,h);
458  w[j]=z;
459  // Rotation can be arbitrary if z = 0.
460  if (z) {
461  z=1.0/z;
462  c=f*z;
463  s=h*z;
464  }
465  f=c*g+s*y;
466  x=c*y-s*g;
467  for (jj=1;jj<=m;jj++) {
468  y=a[jj][j];
469  z=a[jj][i];
470  a[jj][j]=y*c+z*s;
471  a[jj][i]=z*c-y*s;
472  }
473  }
474  rv1[l]=0.0;
475  rv1[k]=f;
476  w[k]=x;
477  }
478  }
479  // free_vector(rv1,1,n);
480  free(rv1);
481 }

484 float_tt pythag(float_tt a, float_tt b) {
485  /* Computes (a 2 + b 2 ) 1=2 without destructive under ow or over ow.
486  */
487  float_tt absa,absb;
488  absa=fabs(a);
489  absb=fabs(b);
490  if (absa > absb)
491  return absa*sqrt(1.0+SQR(absb/absa));
492  else
493  return (absb == 0.0 ? 0.0 : absb*sqrt(1.0+SQR(absa/absb)));
494 }
