341 void CExperimentBase::Propagate(WavePtr wave, float_tt dz)
342 {
343  int ixa, iya;
344  float_tt wr, wi, tr, ti;
345  float_tt scale,t;
346  float_tt dzs=0;
347 
348  float_tt dx, dy;
349  unsigned nx, ny, px;
350 
351  wave->GetResolution(dx, dy);
352  wave->GetSizePixels(nx, ny);
353 
354  complex_tt *w=wave->GetWavePointer();
355 
356  px=nx*ny;
357 
358  // TODO: this will not be thread safe, since m_propxr will be shared amongst threads
359  if (dz != dzs) {
360  dzs = dz;
361  scale = dz*PI;
362 
363  for( ixa=0; ixa<nx; ixa++) {
364  t = scale * (wave->GetKX2(ixa)*wave->GetWavelength());
365  m_propxr[ixa] = (float_tt) cos(t);
366  m_propxi[ixa] = (float_tt) -sin(t);
367  }
368  for( iya=0; iya<ny; iya++) {
369 
370  t = scale * (wave->GetKY2(iya)*wave->GetWavelength());
371  m_propyr[iya] = (float_tt) cos(t);
372  m_propyi[iya] = (float_tt) -sin(t);
373  }
374 
375  }
376  /* end of: if dz != dzs */
377  /*************************************************************/
378 
379  /*************************************************************
380  * Propagation
381  ************************************************************/
382  for (unsigned i=0; i<px; i++)
383  {
384  ixa=i%nx;
385  iya=i/nx;
386  if( wave->GetKX2(ixa) < wave->GetK2Max() ) {
387  if( (wave->GetKX2(ixa) + wave->GetKY2(iya)) < wave->GetK2Max() ) {
388  wr = w[i][0];
389  wi = w[i][1];
390  tr = wr*m_propyr[iya] - wi*m_propyi[iya];
391  ti = wr*m_propyi[iya] + wi*m_propyr[iya];
392  w[i][0] = tr*m_propxr[ixa] - ti*m_propxi[ixa];
393  w[i][1] = tr*m_propxi[ixa] + ti*m_propxr[ixa];
394  } else
395  w[i][0] = w[i][1] = 0.0F;
396  } /* end for(iy..) */
397 
398  else w[i][0] = w[i][1] = 0.0F;
399  } /* end for(ix..) */
400 } /* end propagate */

402 /*------------------------ transmit() ------------------------*/
403 /*
404 transmit the wavefunction thru one layer
405 (simply multiply wave by transmission function)
406 
407 waver,i[ix][iy] = real and imaginary parts of wavefunction
408 transr,i[ix][iy] = real and imag parts of transmission functions
409 
410 nx, ny = size of array
411 
412 on entrance waver,i and transr,i are in real space
413 
414 only waver,i will be changed by this routine
415 */
416 void CExperimentBase::Transmit(WavePtr wave, unsigned sliceIdx) {
417  double wr, wi, tr, ti;
418 
419  complex_tt *w;
420  unsigned nx, ny;
421 
422  w = wave->GetWavePointer();
423  wave->GetSizePixels(nx, ny);
424 
425  /* trans += posx; */
426  for(unsigned ix=0; ix<nx; ix++) for(unsigned iy=0; iy<ny; iy++) {
427  unsigned offset=ix+nx*iy;
428  complex_tt t = m_potential->GetSlicePixel(sliceIdx, ix+m_iPosX, iy+m_iPosY);
429 
430  wr = w[offset][0];
431  wi = w[offset][1];
432  tr = t[0];
433  ti = t[1];
434  w[offset][0] = wr*tr - wi*ti;
435  w[offset][1] = wr*ti + wi*tr;
436  } /* end for(iy.. ix .) */
437 } /* end transmit() */
