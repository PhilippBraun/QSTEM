49 void C2DPotential::AtomBoxLookUp(complex_tt &sum, int Znum, float_tt x, float_tt y, float_tt z, float_tt B)
50 {
51  float_tt dx, dy;
52  int ix, iy;
53 
54  // does the atom box lookup or calculation
55  CPotential::AtomBoxLookUp(sum, Znum, x, y, z, B);
56 
57  /***************************************************************
58  * Do the trilinear interpolation
59  */
60  sum[0] = 0.0;
61  sum[1] = 0.0;
62  if (x*x+y*y+z*z > m_atomRadius2) {
63  return;
64  }
65  x = fabs(x);
66  y = fabs(y);
67  ix = (int)(x/m_ddx);
68  iy = (int)(y/m_ddy);
69  dx = x-(float_tt)ix*m_ddx;
70  dy = y-(float_tt)iy*m_ddy;
71 
72  if ((dx < 0) || (dy<0) ) {
73  /* printf("Warning, dx(%g), dy(%g), dz(%g) < 0, (x=%g, y=%g, z=%g)\n",dx,dy,dz,x,y,z);
74  */
75  if (dx < 0) dx = 0.0;
76  if (dy < 0) dy = 0.0;
77  }
78 
79  if (m_atomBoxes[Znum]->B > 0) {
80  sum[0] = (1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[0][ix][iy][0]+
81  dx*m_atomBoxes[Znum]->potential[0][ix+1][iy][0])+
82  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[0][ix][iy+1][0]+
83  dx*m_atomBoxes[Znum]->potential[0][ix+1][iy+1][0]);
84  sum[1] = (1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->potential[0][ix][iy][1]+
85  dx*m_atomBoxes[Znum]->potential[0][ix+1][iy][1])+
86  dy*((1.0-dx)*m_atomBoxes[Znum]->potential[0][ix][iy+1][1]+
87  dx*m_atomBoxes[Znum]->potential[0][ix+1][iy+1][1]);
88  }
89  else {
90  sum[0] = (1.0-dy)*((1.0-dx)*m_atomBoxes[Znum]->rpotential[0][ix][iy]+
91  dx*m_atomBoxes[Znum]->rpotential[0][ix+1][iy])+
92  dy*((1.0-dx)*m_atomBoxes[Znum]->rpotential[0][ix][iy+1]+
93  dx*m_atomBoxes[Znum]->rpotential[0][ix+1][iy+1]);
94  }
95 }

123 void C2DPotential::_AddAtomRealSpace(std::vector<atom>::iterator &atom,
124  float_tt atomBoxX, unsigned int ix,
125  float_tt atomBoxY, unsigned int iy,
126  float_tt atomZ, unsigned int iAtomZ)
127 {
128  complex_tt dPot;
129  // Coordinates in the total potential space
130  unsigned iz;
131 
132 
133  if (!m_periodicZ) {
134  if (iAtomZ < 0) return;
135  if (iAtomZ >= m_nslices) return;
136  }
137  iz = (iAtomZ+32*m_nslices) % m_nslices; /* shift into the positive range */
138  // x, y are the coordinates in the space of the atom box
139  AtomBoxLookUp(dPot,atom->Znum,atomBoxX,atomBoxY,0, m_tds ? 0 : atom->dw);
140  float_tt atomBoxZ = (double)(iAtomZ+1)*m_cz[0]-atomZ;
141 
142  unsigned idx=ix*m_ny+iy;
143 
144  /* split the atom if it is close to the top edge of the slice */
145  if ((atomBoxZ<0.15*m_cz[0]) && (iz >0)) {
146  m_trans[iz][idx] += 0.5*dPot;
147  m_trans[iz-1][idx] += 0.5*dPot;
148  }
149  /* split the atom if it is close to the bottom edge of the slice */
150  else {
151  if ((atomBoxZ>0.85*m_cz[0]) && (iz < m_nslices-1)) {
152  m_trans[iz][idx] += 0.5*dPot;
153  m_trans[iz+1][idx] += 0.5*dPot;
154  }
155  else {
156  m_trans[iz][idx] += dPot;
157  }
158  }
