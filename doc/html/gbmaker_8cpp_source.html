238 int removeVacancies(atom *atoms,int natoms) {
239  int natomsFinal;
240  int i,i2,j,jz;
241  float_tt totOcc,lastOcc, choice;
242  long idum = -(long)time(NULL);
243  int printLevel = 1;
244 
245 
246  // printf("Time: %d\n",time(NULL));
247  natomsFinal = natoms;
248  qsort((void *)atoms,natoms,sizeof(atom),atomCompareZYX);
249 
250  for(jz = 0,i=0;i<natoms;i++) {
251  if (atoms[i].Znum > 0) {
252  totOcc = atoms[i].occ;
253  for (j=i+1;j<natoms;j++) {
254  // if there is anothe ratom that comes close to within 0.1*sqrt(3) A we will increase
255  // the total occupany and the counter j.
256  if ((fabs(atoms[i].x-atoms[j].x) < 0.1) && (fabs(atoms[i].y-atoms[j].y) < 0.1) && (fabs(atoms[i].z-atoms[j].z) < 0.1)) {
257  totOcc += atoms[j].occ;
258  }
259  else break;
260  } // j-loop
261  // if we encountered atoms in the same position, or the occupancy of the current atom is not 1, then
262  // do something about it:
263  // printf("%d: tocc = %g\n",i,totOcc);
264  if ((totOcc < 1) || (j > i+1)) { // found atoms at equal positions!
265  // ran1 returns a uniform random deviate between 0.0 and 1.0 exclusive of the endpoint values.
266  //
267  // if the total occupancy is less than 1 -> make sure we keep this
268  // if the total occupancy is greater than 1 (unphysical) -> rescale all partial occupancies!
269  if (totOcc < 1.0) choice = ran1(&idum);
270  else choice = totOcc*ran1(&idum);
271  lastOcc = 0;
272  for (i2=i;i2<j;i2++) {
273  // if choice does not match the current atom:
274  // choice will never be 0 or 1(*totOcc)
275  if ((choice <lastOcc) || (choice >=lastOcc+atoms[i2].occ)) {
276  atoms[i2].Znum = 0; // vacancy
277  jz++;
278  }
279  lastOcc += atoms[i2].occ;
280  // if we keep this atom!
281  atoms[i2].occ = 1.0; // to avoid this atom being reduced in occupancy again when reading the cfg file
282  }
283  }
284  i = j-1;
285  }
286  }
287  if ((jz > 0 ) &&(printLevel)) printf("Removed %d atoms because of occupancies < 1 or multiple atoms in the same place\n",jz);
288 
289  // end of managing atoms at equal position
291 
292  natomsFinal = natoms-jz;
293  // printf("%d atoms left\n",natomsFinal);
294  // We now need to move all the atoms with zero Znum to the very end.
295  if (jz > 0) {
296  qsort((void *)atoms,natoms,sizeof(atom),atomCompareZnum);

855 void makeAmorphous() {
856  int g,p,iatom,ix,iy,iz,ic,atomCount = 0,amorphSites,amorphAtoms;

864  int nx,ny,nz;

866 
867 
868  if (Mm == NULL) {

874  /********************************************************
875  * if this grain is an amorphous one ...
876  */
877  if (grains[g].amorphFlag == AMORPHOUS) {
878  r = grains[g].rmin/grains[g].rFactor;
879  /* create an hexagonally closed packed unit cell for initial amorphous structure
880  * The length of each vector is now 1
881  */
882  axCell[0] = r; axCell[1] = 0; axCell[2] = 0;
883  byCell[0] = 0.5*r; byCell[1] = 0.5*sqrt(3.0)*r; byCell[2] = 0;
884  czCell[0] = 0.5*r; czCell[1] = 0.5/sqrt(3.0)*r; czCell[2] = sqrt(5.0)/6*r;
885  /* size of rectangular cell containing 4 atoms: */
886  rCellx = r; rCelly = sqrt(3.0)*r; rCellz = (sqrt(5.0)*r)/3.0;
887 
888  /* determine number of unit cells in super cell */
889  nx = (int)(superCell.ax/rCellx);
890  ny = (int)(superCell.by/rCelly);
891  nz = (int)(superCell.cz/rCellz);
892  amorphSites = 4*nx*ny*nz;
893  amorphCell = (atom *)malloc((amorphSites+1)*sizeof(atom));
894 
895  atomCount = 0;
896  for (ix=0;ix<=nx;ix++) {
897  for (iy=0;iy<=ny;iy++) {
898  for (iz=0;iz<=nz;iz++) {
899  for (ic=0;ic<4;ic++) {
900  /* check if this atom and any of the 4 atoms per rect. unit cell lie within the super cell
901  */
902  amorphCell[atomCount].x = ix*rCellx-(ic==3)*axCell[0]+(ic % 2)*byCell[0]+(ic>1)*czCell[0];
903  amorphCell[atomCount].y = iy*rCelly-(ic==3)*axCell[1]+(ic % 2)*byCell[1]+(ic>1)*czCell[1];
904  amorphCell[atomCount].z = iz*rCellz-(ic==3)*axCell[2]+(ic % 2)*byCell[2]+(ic>1)*czCell[2];
905  if ((amorphCell[atomCount].x >= 0) &&
906  (amorphCell[atomCount].x < superCell.ax) &&
907  (amorphCell[atomCount].y >= 0) &&
908  (amorphCell[atomCount].y < superCell.by) &&
909  (amorphCell[atomCount].z >= 0) &&
910  (amorphCell[atomCount].z < superCell.cz)) {
911  for (p=0;p<grains[g].nplanes;p++) {
912  d = findLambda(grains[g].planes+p,&(amorphCell[atomCount].z),-1);
913  if (d < 0)
914  break;
915  }
916  /* if all the previous test have been successful, this atom is IN */
917  if (p == grains[g].nplanes) atomCount++;
918  }
919  } /* ic ... */
920  } /* iz ... */
921  } /* iy ... */
922  } /* ix ... */

924  /****************************************************************************************
925  * Now we have all the sites within the bounding planes on which we can put atoms
926  */
927  /* the true number of amorphous atoms is # of sites / rFactor^3 */
928  amorphAtoms = (int)floor((float_tt)amorphSites/pow(grains[g].rFactor,3.0));
929  if (amorphAtoms > amorphSites) amorphAtoms = amorphSites;
930  randCount = amorphSites;
931  atomCount = superCell.natoms;
932 
933 
934  superCell.atoms = (atom *)realloc(superCell.atoms,(atomCount+amorphAtoms+1)*sizeof(atom));
935  if (superCell.atoms == NULL) {
936  printf("makeAmorphous: Could not allocate memory for additional atoms!\n");
937  exit(0);
938  }
939 
940  randArray = (int *)malloc(amorphSites*sizeof(int));
941  for (ix=0;ix<amorphSites;ix++) randArray[ix] = ix;
942  /*
943  printf("memory allocation: sC.atoms: %d .. %d, rArray: %d .. %d\n",
944  (int)superCell.atoms,(int)superCell.atoms+(atomCount+amorphAtoms+1)*sizeof(atom),
945  (int)randArray,(int)randArray+amorphSites*sizeof(int));
946  */
947 
948  for (ix=amorphAtoms;ix>0;ix--) {
949  do {
950  iy = (int)((float_tt)rand()*(float_tt)(randCount-1)/(float_tt)(RAND_MAX));
951  if (iy >= randCount) iy = randCount-1;
952  if (iy < 0) iy = 0;
953  // printf("%5d, iy: %d, sites: %d, atoms: %d ",ix,iy,randCount,amorphAtoms);
954  iz = randArray[iy];
955  if (iz > amorphSites) {
956  printf("%5d, iy: %d, sites: %d, atoms: %d ",ix,iy,randCount,amorphAtoms);
957  printf("iz: %d (%d)\n",iz,(int)superCell.atoms[atomCount].z);
958  printf("makeAmorphous: Error because of overlapping memory areas!\n");
959  for (iz=0;iz<=amorphAtoms;iz++)
960  printf("iz=%d: %d\n",iz,randArray[iz]);
961  exit(0);
962  }
963  } while (iz > amorphSites);
964 
965  /* replace already choosen sites with unused ones, so that we don't occupy
966  * any site twice
967  */
968  if (iy == randCount-1) randCount--;
969  else
970  randArray[iy] = randArray[--randCount];
971 
972 
973  iatom = ix % grains[g].natoms;
974  superCell.atoms[atomCount].q = grains[g].unitCell[iatom].q;
975  superCell.atoms[atomCount].dw = grains[g].unitCell[iatom].dw;
976  superCell.atoms[atomCount].occ = grains[g].unitCell[iatom].occ;
977  superCell.atoms[atomCount].Znum = grains[g].unitCell[iatom].Znum;
978  superCell.atoms[atomCount].x = amorphCell[iz].x;
979  superCell.atoms[atomCount].y = amorphCell[iz].y;
980  superCell.atoms[atomCount].z = amorphCell[iz].z;
981  atomCount++;
982  }
