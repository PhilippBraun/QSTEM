558 % function f=Gaussian(sigma,x0,x,[y])
559 % This computes a normal distribution with a Given variance sigma^2
560 
561 function f=Gaussian(sigma,x0,x,y)
562 % decide whether we do this in 1, or 2 dimensions:
563 [Ny,Nx] = size(x);
564 if (Nx==1) || (Ny==1)
565  dx = x(2)-x(1);
566  f = dx/(sigma*sqrt(2*pi))*exp(-(x-x0).^2./(2*sigma^2));
567 else
568  dx = max(x(2,1)-x(1,1),x(1,2)-x(1,1));
569  dy = max(y(2,1)-y(1,1),y(1,2)-y(1,1));
570  f = dx*dy/(sigma^2*2*pi)*exp(-((x-x0(1)).^2+(y-x0(2)).^2)./(2*sigma^2));
571 end
