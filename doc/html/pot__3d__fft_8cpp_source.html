52  float_tt atomBoxY, unsigned int iAtomY,

60  unsigned iax1 = iAtomX+m_iRadX >= m_nx ? m_nx-1 : iAtomX+m_iRadX;
61  unsigned iay0 = iAtomY-m_iRadY < 0 ? 0 : iAtomY-m_iRadY;
62  unsigned iay1 = iAtomY+m_iRadY >= m_ny ? m_ny-1 : iAtomY+m_iRadY;
63  // if within the potential map range:
64  if ((iax0 < m_nx) && (iax1 >= 0) && (iay0 < m_ny) && (iay1 >= 0)) {
65  // define range of sampling from atomZ-/+atomRadius
66  int iaz0 = iAtomZ-m_iRadZ < 0 ? -iAtomZ : -m_iRadZ;
67  unsigned iaz1 = iAtomZ+m_iRadZ >= m_nslices ? m_nslices-iAtomZ-1 : m_iRadZ;

69  // retrieve the pointer for this atom
70  ComplexVector atPotPtr;

72 #if USE_Q_POT_OFFSETS

77 #endif // USE_Q_POT_OFFSETS

79  int iOffsLimLo = -Nr*(Nz_lut-1);
80  unsigned iOffsStep = nzSub*Nr;
81 
82  // Slices around the slice that this atom is located in must be affected by this atom:
83  // iaz must be relative to the first slice of the atom potential box.
84  for (unsigned iax=iax0; iax <= iax1; iax++) {
85  unsigned idx = iax*m_ny+iay0;
86  float_tt *potPtr = &(m_trans[iAtomZ+iaz0][idx][0]);
87 
89  // Computation of Radius must be made faster by using pre-calculated ddx
90  // and LUT for sqrt:
91  // use of sqrt slows down from 120sec to 180 sec.
92  float_tt x2 = iax*m_dx - atomBoxX; x2 *= x2;
93  for (unsigned iay=iay0; iay <= iay1; iay++) {
94  // printf("iax=%d, iay=%d\n",iax,iay);
95  float_tt y2 = iay*m_dy - atomBoxY; y2 *= y2;
96  float_tt r = sqrt(x2+y2);
97  // r = (x2+y2);
98  float_tt ddr = r/m_dr;
99  unsigned ir = (int)floor(ddr);
100  // add in different slices, once r has been defined
101  if (ir < Nr-1) {
102  ddr = ddr-(double)ir;
103  float_tt *ptr = potPtr;
104 
105  float_tt dOffsZ = (iAtomZ+iaz0-atomZ/m_sliceThickness)*nzSub;
106 #if Z_INTERPOLATION
107  unsigned iOffsZ = (unsigned)dOffsZ;

109 #else
110  unsigned iOffsZ = (int)(dOffsZ+0.5);
111 #endif
112  iOffsZ *= Nr;
113

119 #if Z_INTERPOLATION

131 #if USE_Q_POT_OFFSETS

135  }
136 #endif // USE_Q_POT_OFFSETS
137 #endif // Z_INTERPOLATION

143  // do the real part by linear interpolation in r-dimension:
144 #if Z_INTERPOLATION

154 #else // Z_INTERPOLATION

156 #if USE_Q_POT_OFFSETS

161 #endif // USE_Q_POT_OFFSETS

163
