41 void CImgReader::ReadHeader(const std::string &filebase, const std::vector<unsigned> &position)
42 {
43  FILE *fp;
44  std::vector<double> params;
45  std::string filename=BuildFilenameString(filebase, position);
46  if ((fp = fopen(filename.c_str(),"rb"))==NULL)
47  {
48  sprintf(m_buf, "Could not open file %s for reading header.\n",filename.c_str());
49  throw std::runtime_error(m_buf);
50  }
51  fread((void*)&m_headerSize, 4, 1,fp);
52  fread((void*)&m_paramSize, 4, 1, fp);
53  fread((void*)&m_commentSize, 4, 1, fp);
54  fread((void*)&m_nx, 4, 1, fp);

56  fread((void*)&m_complexFlag, 4, 1, fp);
57  fread((void*)&m_dataSize, 4, 1, fp);
58  fread((void*)&m_version, 4, 1, fp);
59  fread((void*)&m_t, 8, 1, fp); // thickness
60  fread((void*)&m_dx, 8, 1, fp); // resolution in X
61  fread((void*)&m_dy, 8, 1, fp); // resolution in Y
62 
63  if (m_paramSize>0)
64  {
65  m_params=std::vector<double>(m_paramSize);
66  fread((void *)&m_params[0],sizeof(double),m_paramSize,fp);
67  }
68  if (m_commentSize>0)
69  {
70  fread((void*)m_buf, 1, m_commentSize, fp);
71  m_comment = std::string(m_buf);
72  m_comment.resize(m_commentSize);
73  }
74 
75  if (fp != NULL) fclose(fp);
76 }

159 void CImgReader::_ReadImageData(const std::string &filebase, const std::vector<unsigned> &position, void *pix)
160 {
161  std::ifstream inputfile;
162 
163  std::string filename=BuildFilenameString(filebase, position);
164  inputfile.open (filename, std::ios::in | std::ios::binary);
165 
166  if (inputfile.good())
167  {
168  // Seek to the location of the actual data
169  //fseek(fp,56+(m_commentSize)+(2*m_paramSize),SEEK_SET);
170  inputfile.seekg(56+(m_commentSize)+(2*m_paramSize));
171 
172  // this is type-agnostic - the type interpretation is done by the
173  // function sending in the pointer. It casts it as char for the reading,
174  // but it then "knows" that it is double, complex, whatever, based on the
175  // type of the data that it passed into this function.
176  //
177  // Complex data is determined/communicated by the m_complexFlag, which is read in the header.
178  // The data size is doubled for complex data.
179  inputfile.read(reinterpret_cast <char*> (pix), m_nx*m_ny*m_dataSize);
180  }
181  else {
182  printf("Could not open file %s for reading\n",filename.c_str());
183  /* wait a short while */
184  //while (nRead < 1e5) nRead++;
185  }
186 
187  inputfile.close();
188 }
