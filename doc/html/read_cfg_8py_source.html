30 def get_atom_arrays(filename):
31  """
32  Returns a dictionary with an entry per element. The key to the dictionary
33  is the atomic number, while the value is an array containing coordinates,
34  Debye-Waller factors, occupancies, and charges on a site-specific basis.
35  """
36  output_dict={}
37  f=open(filename)
38  txt=f.read()
39  f.close()
40  stripped_sections = array_re.findall(txt)
41  for element in stripped_sections:
42  # interprets the extracted string as a numpy array, and reshapes it
43  # so that we end up with one atomic coordinate per
44  output_dict[element[0]]=np.fromstring(element[2], sep=' ').reshape((-1,6))
45  return output_dict
