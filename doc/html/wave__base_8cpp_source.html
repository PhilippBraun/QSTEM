135  void CBaseWave::InitializeKVectors()
136  {
137  m_kx.resize(m_nx);
138  m_kx2.resize(m_nx);
139  m_ky.resize(m_ny);
140  m_ky2.resize(m_ny);
141 
142  float_tt ax = m_dx*m_nx;
143  float_tt by = m_dy*m_ny;
144  for(unsigned ixa=0; ixa<m_nx; ixa++)
145  {
146  m_kx[ixa] = (ixa>m_nx/2) ? (float_tt)(ixa-m_nx)/ax :
147  (float_tt)ixa/ax;
148  m_kx2[ixa] = m_kx[ixa]*m_kx[ixa];
149  }
150  for(unsigned iya=0; iya<m_ny; iya++) {
151  m_ky[iya] = (iya>m_ny/2) ?
152  (float_tt)(iya-m_ny)/by :
153  (float_tt)iya/by;
154  m_ky2[iya] = m_ky[iya]*m_ky[iya];
155  }
156  m_k2max = m_nx/(2.0F*ax);
157  if (m_ny/(2.0F*by) < m_k2max ) m_k2max = m_ny/(2.0F*by);
158  m_k2max = 2.0/3.0 * m_k2max;
159  m_k2max = m_k2max*m_k2max;
160 }
