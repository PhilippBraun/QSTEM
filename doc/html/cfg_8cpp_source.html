107 int CCfgReader::ReadNextAtom(atom *newAtom) {
108  unsigned element;
109  char *str = NULL;
110  float_tt mass;
111  char buf[NCMAX];
112 
113  if (m_fp == NULL) {
114  printf("Invalid CFG file!\n");
115  return -1;
116  }
117  //resetParamFile(m_fp);
118 
119  if (fgets(buf,NCMAX,m_fp) == NULL) return -1;
120  /* check, if this is a new mass number */
121  str = strnext(buf," \t");
122  if ((atof(buf) >= 1.0) && ((str==NULL) || (*str == '#'))) {
123  mass = atof(buf);
124  // printf("nV: %d, eC: %d (%g)\n",noVelocityFlag, entryCount,atof(m_buf));
125  if (fgets(buf,NCMAX,m_fp) == NULL) return -1;
126  element = getZNumber(buf);
127  // printf("*** found element %d (%s %d) ***\n",element,m_buf,strlen(m_buf));
128  if (fgets(buf,NCMAX,m_fp) == NULL) return -1;
129  }
130  str = buf;
131  // skip leading spaces:
132  while (strchr(" \t",*str) != NULL) str++;
133  for (unsigned j=0;j<m_entryCount;j++) {
134  if (str==NULL) {
135  printf("readNextCFGatom: Error: incomplete data line: >%s<\n",buf);
136  return -1;
137  }
138  m_atomData[j] = atof(str); str=strnext(str," \t");
139  }
140 
141  newAtom->x = m_atomData[0];
142  newAtom->y = m_atomData[1];
143  newAtom->z = m_atomData[2];
144  // Temporary initialization: dw, occ, q are read if they exist
145  newAtom->dw = 0.45*28.0/mass;
146  newAtom->occ = 1.0;
147  newAtom->q = 0.0;
148 
149  newAtom->Znum = element;
150  newAtom->mass = mass;
151  // read the DW-factor
152  if (m_entryCount > 3+3*(1-(int)m_noVelocity))
153  newAtom->dw = m_atomData[3+3*(1-(int)m_noVelocity)];
154  // read the atom's occupancy:
155  if (m_entryCount > 4+3*(1-(int)m_noVelocity))
156  newAtom->occ = m_atomData[4+3*(1-(int)m_noVelocity)];
157  // read the atom's charge:
158  if (m_entryCount > 5+3*(1-(int)m_noVelocity))
159  newAtom->q = m_atomData[5+3*(1-(int)m_noVelocity)];
160  // printf("Atom: %d (%g %g %g), occ=%g, q=%g\n",newAtom->Znum,newAtom->x,newAtom->y,newAtom->z,newAtom->occ,newAtom->q);
161 
162  return 0;
163 }
