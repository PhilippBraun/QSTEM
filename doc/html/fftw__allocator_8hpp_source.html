442 template <class R>
443 inline FFTWComplex<R> & operator /=(FFTWComplex<R> &a, const FFTWComplex<R> &b) {
444  typename FFTWComplex<R>::value_type sm = b.squaredMagnitude();
445  typename FFTWComplex<R>::value_type t = (a.re()*b.re()+a.im()*b.im())/sm;
446  a.im() = (b.re()*a.im()-a.re()*b.im())/sm;
447  a.re() = t;
448  return a;
