208 ANTLR3_INLINE static pANTLR3_COMMON_TOKEN
209 nextTokenStr (pANTLR3_TOKEN_SOURCE toksource)
210 {
211  pANTLR3_LEXER lexer;
212  pANTLR3_RECOGNIZER_SHARED_STATE state;
213  pANTLR3_INPUT_STREAM input;
214  pANTLR3_INT_STREAM istream;
215 
216  lexer = (pANTLR3_LEXER)(toksource->super);
217  state = lexer->rec->state;
218  input = lexer->input;
219  istream = input->istream;
220 
223  for (;;)
224  {
225  // Get rid of any previous token (token factory takes care of
226  // any de-allocation when this token is finally used up.
227  //
228  state->token = NULL;
229  state->error = ANTLR3_FALSE; // Start out without an exception
230  state->failed = ANTLR3_FALSE;
231 
232  // Now call the matching rules and see if we can generate a new token
233  //
234  for (;;)
235  {
236  // Record the start of the token in our input stream.
237  //
238  state->channel = ANTLR3_TOKEN_DEFAULT_CHANNEL;
239  state->tokenStartCharIndex = (ANTLR3_MARKER)(((pANTLR3_UINT8)input->nextChar));
240  state->tokenStartCharPositionInLine = input->charPositionInLine;
241  state->tokenStartLine = input->line;
242  state->text = NULL;
243  if (istream->_LA(istream, 1) == ANTLR3_CHARSTREAM_EOF)
244  {
245  // Reached the end of the current stream, nothing more to do if this is
246  // the last in the stack.
247  //
248  pANTLR3_COMMON_TOKEN teof = &(toksource->eofToken);
249 
250  teof->setStartIndex (teof, lexer->getCharIndex(lexer));
251  teof->setStopIndex (teof, lexer->getCharIndex(lexer));
252  teof->line = lexer->getLine(lexer);
253  teof->factoryMade = ANTLR3_TRUE; // This isn't really manufactured but it stops things from trying to free it
254  return teof;
255  }
256 
257  state->token = NULL;
258  state->error = ANTLR3_FALSE; // Start out without an exception
259  state->failed = ANTLR3_FALSE;
260 
261  // Call the generated lexer, see if it can get a new token together.
262  //
263  lexer->mTokens(lexer->ctx);
264 
265  if (state->error == ANTLR3_TRUE)
266  {
267  // Recognition exception, report it and try to recover.
268  //
269  state->failed = ANTLR3_TRUE;
270  lexer->rec->reportError(lexer->rec);
271  lexer->recover(lexer);
272  }
273  else
274  {
275  if (state->token == NULL)
276  {
277  // Emit the real token, which adds it in to the token stream basically
278  //
279  emit(lexer);
280  }
281  else if (state->token == &(toksource->skipToken))
282  {
283  // A real token could have been generated, but "Computer say's naaaaah" and it
284  // it is just something we need to skip altogether.
285  //
286  continue;
287  }
288 
289  // Good token, not skipped, not EOF token
290  //
291  return state->token;
292  }
293  }
294  }

316 static pANTLR3_COMMON_TOKEN
317 nextToken (pANTLR3_TOKEN_SOURCE toksource)
318 {
319  pANTLR3_COMMON_TOKEN tok;
320 
321  // Find the next token in the current stream
322  //
323  tok = nextTokenStr(toksource);
324 
325  // If we got to the EOF token then switch to the previous
326  // input stream if there were any and just return the
327  // EOF if there are none. We must check the next token
328  // in any outstanding input stream we pop into the active
329  // role to see if it was sitting at EOF after PUSHing the
330  // stream we just consumed, otherwise we will return EOF
331  // on the reinstalled input stream, when in actual fact
332  // there might be more input streams to POP before the
333  // real EOF of the whole logical inptu stream. Hence we
334  // use a while loop here until we find somethign in the stream
335  // that isn't EOF or we reach the actual end of the last input
336  // stream on the stack.
337  //
338  while (tok->type == ANTLR3_TOKEN_EOF)
339  {
340  pANTLR3_LEXER lexer;
341 
342  lexer = (pANTLR3_LEXER)(toksource->super);
343 
344  if (lexer->rec->state->streams != NULL && lexer->rec->state->streams->size(lexer->rec->state->streams) > 0)
345  {
346  // We have another input stream in the stack so we
347  // need to revert to it, then resume the loop to check
348  // it wasn't sitting at EOF itself.
349  //
350  lexer->popCharStream(lexer);
351  tok = nextTokenStr(toksource);
352  }
353  else
354  {
355  // There were no more streams on the input stack
356  // so this EOF is the 'real' logical EOF for
357  // the input stream. So we just exit the loop and
358  // return the EOF we have found.
359  //
360  break;
361  }
362 
363  }
364 
365  // return whatever token we have, which may be EOF
366  //
367  return tok;
368 }

462  (ANTLR3_UINT32)(lexer->rec->state->tokenStartLine),
463  (ANTLR3_UINT32)(lexer->rec->state->tokenStartCharPositionInLine)
464  );
465  width = ANTLR3_UINT32_CAST(((pANTLR3_UINT8)(lexer->input->data)+(lexer->input->size(lexer->input))) - (pANTLR3_UINT8)(lexer->rec->state->tokenStartCharIndex));
466 
467  if (width >= 1)
468  {
469  ANTLR3_FPRINTF(stderr, "looks like this:\n\t\t%.*s\n", width > 20 ? 20 : width ,(pANTLR3_UINT8)(lexer->rec->state->tokenStartCharIndex));
470  }
471  else
472  {
473  ANTLR3_FPRINTF(stderr, "is also the end of the line, so you must check your lexer rules\n");
474  }
475  }

479 static void setCharStream (pANTLR3_LEXER lexer, pANTLR3_INPUT_STREAM input)
480 {
481  /* Install the input interface
482  */
483  lexer->input = input;
484 
485  /* We may need a token factory for the lexer; we don't destroy any existing factory
486  * until the lexer is destroyed, as people may still be using the tokens it produced.
487  * TODO: Later I will provide a dup() method for a token so that it can extract itself
488  * out of the factory.
489  */
490  if (lexer->rec->state->tokFactory == NULL)
491  {
492  lexer->rec->state->tokFactory = antlr3TokenFactoryNew(input);
493  }
494  else
495  {
496  /* When the input stream is being changed on the fly, rather than
497  * at the start of a new lexer, then we must tell the tokenFactory
498  * which input stream to adorn the tokens with so that when they
499  * are asked to provide their original input strings they can
500  * do so from the correct text stream.
501  */
502  lexer->rec->state->tokFactory->setInputStream(lexer->rec->state->tokFactory, input);
503  }
504 
505  /* Propagate the string factory so that we preserve the encoding form from
506  * the input stream.
507  */
508  if (lexer->rec->state->tokSource->strFactory == NULL)
509  {
510  lexer->rec->state->tokSource->strFactory = input->strFactory;
511 
512  // Set the newly acquired string factory up for our pre-made tokens
513  // for EOF.
514  //
515  if (lexer->rec->state->tokSource->eofToken.strFactory == NULL)
516  {
517  lexer->rec->state->tokSource->eofToken.strFactory = input->strFactory;
518  }
519  }
520 
521  /* This is a lexer, install the appropriate exception creator
522  */
523  lexer->rec->exConstruct = antlr3RecognitionExceptionNew;
524 
525  /* Set the current token to nothing
526  */
527  lexer->rec->state->token = NULL;
528  lexer->rec->state->text = NULL;
529  lexer->rec->state->tokenStartCharIndex = -1;
530 
531  /* Copy the name of the char stream to the token source
532  */
533  lexer->rec->state->tokSource->fileName = input->fileName;
534 }

550 static void
551 pushCharStream (pANTLR3_LEXER lexer, pANTLR3_INPUT_STREAM input)
552 {
553  // Do we need a new input stream stack?
554  //
555  if (lexer->rec->state->streams == NULL)
556  {
557  // This is the first call to stack a new
558  // stream and so we must create the stack first.
559  //
560  lexer->rec->state->streams = antlr3StackNew(0);
561 
562  if (lexer->rec->state->streams == NULL)
563  {
564  // Could not do this, we just fail to push it.
565  // TODO: Consider if this is what we want to do, but then
566  // any programmer can override this method to do something else.
567  return;
568  }
569  }
570 
571  // We have a stack, so we can save the current input stream
572  // into it.
573  //
574  lexer->input->istream->mark(lexer->input->istream);
575  lexer->rec->state->streams->push(lexer->rec->state->streams, lexer->input, NULL);
576 
577  // And now we can install this new one
578  //
579  lexer->setCharStream(lexer, input);
580 }

596 static void
597 popCharStream (pANTLR3_LEXER lexer)
598 {
599  pANTLR3_INPUT_STREAM input;
600 
601  // If we do not have a stream stack or we are already at the
602  // stack bottom, then do nothing.
603  //
604  if (lexer->rec->state->streams != NULL && lexer->rec->state->streams->size(lexer->rec->state->streams) > 0)
605  {
606  // We just leave the current stream to its fate, we do not close
607  // it or anything as we do not know what the programmer intended
608  // for it. This method can always be overridden of course.
609  // So just find out what was currently saved on the stack and use
610  // that now, then pop it from the stack.
611  //
612  input = (pANTLR3_INPUT_STREAM)(lexer->rec->state->streams->top);
613  lexer->rec->state->streams->pop(lexer->rec->state->streams);
614 
615  // Now install the stream as the current one.
616  //
617  lexer->setCharStream(lexer, input);
618  lexer->input->istream->rewindLast(lexer->input->istream);
619  }
620  return;
621 }

628 static pANTLR3_COMMON_TOKEN
629 emit (pANTLR3_LEXER lexer)
630 {
631  pANTLR3_COMMON_TOKEN token;
632 
633  /* We could check pointers to token factories and so on, but
634  * we are in code that we want to run as fast as possible
635  * so we are not checking any errors. So make sure you have installed an input stream before
636  * trying to emit a new token.
637  */
638  token = lexer->rec->state->tokFactory->newToken(lexer->rec->state->tokFactory);
639 
640  /* Install the supplied information, and some other bits we already know
641  * get added automatically, such as the input stream it is associated with
642  * (though it can all be overridden of course)
643  */
644  token->type = lexer->rec->state->type;
645  token->channel = lexer->rec->state->channel;
646  token->start = lexer->rec->state->tokenStartCharIndex;
647  token->stop = lexer->getCharIndex(lexer) - 1;
648  token->line = lexer->rec->state->tokenStartLine;
649  token->charPosition = lexer->rec->state->tokenStartCharPositionInLine;
650 
651  if (lexer->rec->state->text != NULL)
652  {
653  token->textState = ANTLR3_TEXT_STRING;
654  token->tokText.text = lexer->rec->state->text;
655  }
656  else
657  {
658  token->textState = ANTLR3_TEXT_NONE;
659  }
660 
661  lexer->rec->state->token = token;
662 
663  return token;
664 }

669 static void
670 freeLexer (pANTLR3_LEXER lexer)
671 {
672  // This may have ben a delegate or delegator lexer, in which case the
673  // state may already have been freed (and set to NULL therefore)
674  // so we ignore the state if we don't have it.
675  //
676  if (lexer->rec->state != NULL)
677  {
678  if (lexer->rec->state->streams != NULL)
679  {
680  lexer->rec->state->streams->free(lexer->rec->state->streams);
681  }
682  if (lexer->rec->state->tokFactory != NULL)
683  {
684  lexer->rec->state->tokFactory->close(lexer->rec->state->tokFactory);
685  lexer->rec->state->tokFactory = NULL;
686  }
687  if (lexer->rec->state->tokSource != NULL)
688  {
689  ANTLR3_FREE(lexer->rec->state->tokSource);
690  lexer->rec->state->tokSource = NULL;
691  }
692  }
693  if (lexer->rec != NULL)
694  {
695  lexer->rec->free(lexer->rec);
696  lexer->rec = NULL;
697  }
698  ANTLR3_FREE(lexer);
699 }
