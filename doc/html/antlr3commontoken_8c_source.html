140 static void
141 newPool(pANTLR3_TOKEN_FACTORY factory)
142 {
143  /* Increment factory count
144  */
145  factory->thisPool++;
146 
147  // If we were reusing this token factory then we may already have a pool
148  // allocated. If we exceeded the max avaible then we must allocate a new
149  // one.
150  if (factory->thisPool > factory->maxPool)
151  {
152  /* Ensure we have enough pointers allocated
153  */
154  factory->pools = (pANTLR3_COMMON_TOKEN *)
155  ANTLR3_REALLOC( (void *)factory->pools, /* Current pools pointer (starts at NULL) */
156  (ANTLR3_UINT32)((factory->thisPool + 1) * sizeof(pANTLR3_COMMON_TOKEN *)) /* Memory for new pool pointers */
157  );
158 
159  /* Allocate a new pool for the factory
160  */
161  factory->pools[factory->thisPool] =
162  (pANTLR3_COMMON_TOKEN)
163  ANTLR3_CALLOC(1, (size_t)(sizeof(ANTLR3_COMMON_TOKEN) * ANTLR3_FACTORY_POOL_SIZE));
164 
165  // We now have a new pool and can track it as the maximum we have created so far
166  //
167  factory->maxPool = factory->thisPool;
168  }
169 
170  /* Reset the counters
171  */
172  factory->nextToken = 0;
173 
174  /* Done
175  */
176  return;
177 }

179 static pANTLR3_COMMON_TOKEN
180 newPoolToken(pANTLR3_TOKEN_FACTORY factory)
181 {
182  pANTLR3_COMMON_TOKEN token;
183 
184  /* See if we need a new token pool before allocating a new
185  * one
186  */
187  if (factory->nextToken >= ANTLR3_FACTORY_POOL_SIZE)
188  {
189  /* We ran out of tokens in the current pool, so we need a new pool
190  */
191  newPool(factory);
192  }
193 
194  /* Assuming everything went well (we are trying for performance here so doing minimal
195  * error checking. Then we can work out what the pointer is to the next token.
196  */
197  token = factory->pools[factory->thisPool] + factory->nextToken;
198  factory->nextToken++;
199 
200  /* We have our token pointer now, so we can initialize it to the predefined model.
201  * We only need do this though if the token is not already initialized, we just check
202  * an api function pointer for this as they are allocated via calloc.
203  */
204  if (token->setStartIndex == NULL)
205  {
206  antlr3SetTokenAPI(token);
207 
208  // It is factory made, and we need to copy the string factory pointer
209  //
210  token->factoryMade = ANTLR3_TRUE;
211  token->strFactory = factory->input == NULL ? NULL : factory->input->strFactory;
212  token->input = factory->input;
213  }
214 
215  /* And we are done
216  */
217  return token;
218 }

230 static void
231 factoryClose (pANTLR3_TOKEN_FACTORY factory)
232 {
233  pANTLR3_COMMON_TOKEN pool;
234  ANTLR3_INT32 poolCount;
235  ANTLR3_UINT32 limit;
236  ANTLR3_UINT32 token;
237  pANTLR3_COMMON_TOKEN check;
238 
239  /* We iterate the token pools one at a time
240  */
241  for (poolCount = 0; poolCount <= factory->thisPool; poolCount++)
242  {
243  /* Pointer to current pool
244  */
245  pool = factory->pools[poolCount];
246 
247  /* Work out how many tokens we need to check in this pool.
248  */
249  limit = (poolCount == factory->thisPool ? factory->nextToken : ANTLR3_FACTORY_POOL_SIZE);
250 
251  /* Marginal condition, we might be at the start of a brand new pool
252  * where the nextToken is 0 and nothing has been allocated.
253  */
254  if (limit > 0)
255  {
256  /* We have some tokens allocated from this pool
257  */
258  for (token = 0; token < limit; token++)
259  {
260  /* Next one in the chain
261  */
262  check = pool + token;
263 
264  /* If the programmer made this a custom token, then
265  * see if we need to call their free routine.
266  */
267 /* if (check->custom != NULL && check->freeCustom != NULL)
268  {
269  check->freeCustom(check->custom);
270  check->custom = NULL;
271  }*/
272  }
273  }
274 
275  /* We can now free this pool allocation
276  */
277  ANTLR3_FREE(factory->pools[poolCount]);
278  factory->pools[poolCount] = NULL;
279  }
280 
281  /* All the pools are deallocated we can free the pointers to the pools
282  * now.
283  */
284  ANTLR3_FREE(factory->pools);
285 
286  /* Finally, we can free the space for the factory itself
287  */
288  ANTLR3_FREE(factory);
289 }

328 static pANTLR3_STRING getText (pANTLR3_COMMON_TOKEN token)
329 {
330  switch (token->textState)
331  {
332  case ANTLR3_TEXT_STRING:
333 
334  // Someone already created a string for this token, so we just
335  // use it.
336  //
337  return token->tokText.text;
338  break;
339 
340  case ANTLR3_TEXT_CHARP:
341 
342  // We had a straight text pointer installed, now we
343  // must convert it to a string. Note we have to do this here
344  // or otherwise setText8() will just install the same char*
345  //
346  if (token->strFactory != NULL)
347  {
348  token->tokText.text = token->strFactory->newStr8(token->strFactory, (pANTLR3_UINT8)token->tokText.chars);
349  token->textState = ANTLR3_TEXT_STRING;
350  return token->tokText.text;
351  }
352  else
353  {
354  // We cannot do anything here
355  //
356  return NULL;
357  }
358  break;
359 
360  default:
361 
362  // EOF is a special case
363  //
364  if (token->type == ANTLR3_TOKEN_EOF)
365  {
366  token->tokText.text = token->strFactory->newStr8(token->strFactory, (pANTLR3_UINT8)"<EOF>");
367  token->textState = ANTLR3_TEXT_STRING;
368  token->tokText.text->factory = token->strFactory;
369  return token->tokText.text;
370  }
371 
372 
373  // We had nothing installed in the token, create a new string
374  // from the input stream
375  //
376 
377  if (token->input != NULL)
378  {
379 
380  return token->input->substr( token->input,
381  token->getStartIndex(token),
382  token->stop
383  );
384  }
385 
386  // Nothing to return, there is no input stream
387  //
388  return NULL;
389  break;
390  }

392 static void setText8 (pANTLR3_COMMON_TOKEN token, pANTLR3_UINT8 text)
393 {
394  // No text to set, so ignore
395  //
396  if (text == NULL) return;
397 
398  switch (token->textState)
399  {
400  case ANTLR3_TEXT_NONE:
401  case ANTLR3_TEXT_CHARP: // Caller must free before setting again, if it needs to be freed
402 
403  // Nothing in there yet, or just a char *, so just set the
404  // text as a pointer
405  //
406  token->textState = ANTLR3_TEXT_CHARP;
407  token->tokText.chars = (pANTLR3_UCHAR)text;
408  break;
409 
410  default:
411 
412  // It was already a pANTLR3_STRING, so just override it
413  //
414  token->tokText.text->set8(token->tokText.text, (const char *)text);
415  break;
416  }
417 
418  // We are done
419  //
420  return;
421 }
