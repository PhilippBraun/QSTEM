79 void C2DFFTPotential::AddAtomNonPeriodic(std::vector<atom>::iterator &atom,
80  float_tt atomBoxX, unsigned int iAtomX,
81  float_tt atomBoxY, unsigned int iAtomY,
82  float_tt atomZ)
83 {
84  unsigned iAtomZ = (int)floor(atomZ/m_sliceThickness);
85  unsigned iax0 = iAtomX-m_iRadX < 0 ? 0 : iAtomX-m_iRadX;
86  unsigned iax1 = iAtomX+m_iRadX >= m_nx ? m_nx-1 : iAtomX+m_iRadX;
87  unsigned iay0 = iAtomY-m_iRadY < 0 ? 0 : iAtomY-m_iRadY;
88  unsigned iay1 = iAtomY+m_iRadY >= m_ny ? m_ny-1 : iAtomY+m_iRadY;
89  // if within the potential map range:
90  if ((iax0 < m_nx) && (iax1 >= 0) && (iay0 < m_ny) && (iay1 >= 0))
91  {
92  float_tt ddx = (-(double)iax0+(atomBoxX/m_dx-(double)m_iRadX))*(double)OVERSAMPLING;
93  float_tt ddy = (-(double)iay0+(atomBoxY/m_dy-(double)m_iRadY))*(double)OVERSAMPLING;
94  unsigned iOffsX = (int)floor(ddx);
95  unsigned iOffsY = (int)floor(ddy);
96  ddx -= (double)iOffsX;
97  ddy -= (double)iOffsY;
98  float_tt s11 = (1-ddx)*(1-ddy);
99  float_tt s12 = (1-ddx)*ddy;
100  float_tt s21 = ddx*(1-ddy);
101  float_tt s22 = ddx*ddy;
102  complex_tt *atPotPtr = GetAtomPotential2D(atom->Znum,m_tds ? 0 : atom->dw);
103 
104  for (unsigned iax=iax0; iax < iax1; iax++)
105  {
106  unsigned idx = iax*m_ny+iay0;
107  // printf("(%d, %d): %d,%d\n",iax,nyAtBox,(iOffsX+OVERSAMPLING*(iax-iax0)),iOffsY+iay1-iay0);
108  // potPtr and ptr are of type (float *)
109  float_tt *potPtr = &(m_trans[iAtomZ][idx][0]);
110  float_tt *ptr = &(atPotPtr[(iOffsX+OVERSAMPLING*(iax-iax0))*m_nyAtBox+iOffsY][0]);
111  for (unsigned iay=iay0; iay < iay1; iay++)
112  {
113  *potPtr += s11*(*ptr)+s12*(*(ptr+2))+s21*(*(ptr+m_nyAtBox2))+s22*(*(ptr+m_nyAtBox2+2));
114 
115  potPtr++;
116  // *potPtr = 0;
117  potPtr++;
118  ptr += 2*OVERSAMPLING;
119  }
120  }
121  }

125 void C2DFFTPotential::AddAtomPeriodic(std::vector<atom>::iterator &atom,
126  float_tt atomBoxX, unsigned int iAtomX,
127  float_tt atomBoxY, unsigned int iAtomY,
128  float_tt atomZ)
129 {
130  unsigned iAtomZ = (int)floor(atomZ/m_sliceThickness);
131  unsigned iax0 = iAtomX-m_iRadX+2*m_nx;
132  unsigned iax1 = iAtomX+m_iRadX+2*m_nx;
133  unsigned iay0 = iAtomY-m_iRadY+2*m_ny;
134  unsigned iay1 = iAtomY+m_iRadY+2*m_ny;
135 
136  float_tt ddx = (-(double)iax0+(atomBoxX/m_dx-(double)(m_iRadX-2*m_nx)))*(double)OVERSAMPLING;
137  float_tt ddy = (-(double)iay0+(atomBoxY/m_dy-(double)(m_iRadY-2*m_ny)))*(double)OVERSAMPLING;
138  unsigned iOffsX = (int)floor(ddx);
139  unsigned iOffsY = (int)floor(ddy);
140  ddx -= (double)iOffsX;
141  ddy -= (double)iOffsY;
142 
143  float_tt s22 = (1-ddx)*(1-ddy);
144  float_tt s21 = (1-ddx)*ddy;
145  float_tt s12 = ddx*(1-ddy);
146  float_tt s11 = ddx*ddy;
147 
148  complex_tt *atPotPtr = GetAtomPotential2D(atom->Znum,m_tds ? 0 : atom->dw);
149 
150  for (unsigned iax=iax0; iax < iax1; iax++) { // TODO: should use ix += OVERSAMPLING
151  // printf("(%d, %d): %d,%d\n",iax,nyAtBox,(iOffsX+OVERSAMPLING*(iax-iax0)),iOffsY+iay1-iay0);
152  // potPtr and ptr are of type (float *)
154  // Only the exact slice that this atom is located in is affected by this atom:
155  int atPosX = (OVERSAMPLING*(iax-iax0)-iOffsX);
156  if ((atPosX >= 0) && (atPosX < m_nyAtBox-1)) {
157  float_tt *ptr = &(atPotPtr[atPosX*m_nyAtBox-iOffsY][0]);
158  for (unsigned iay=iay0; iay < iay1; iay++) {
159  // wrap around when end of y-line is reached:
160  int atPosY = (iay-iay0)*OVERSAMPLING-iOffsY;
161  if ((atPosY < m_nyAtBox-1) && (atPosY >=0)) {
162  // do the real part
163  unsigned idx = (iax % m_nx)*m_ny + (iay % m_ny);
164  m_trans[iAtomZ][idx][0] +=
165  s11*(*ptr)+s12*(*(ptr+2))+s21*(*(ptr+m_nyAtBox2))+s22*(*(ptr+m_nyAtBox2+2));
166  }
167  // make imaginary part zero for now
168  // *potPtr = 0; potPtr++;
169  ptr += 2*OVERSAMPLING;
170  }
171  } // if atPosX within limits
172  }
173 }
